12pub struct InitializedDb {
13    /// Pool used to interact with the DB
14    pub pool: SplitPool,
15    /// The CRSQL clock
16    pub clock: Arc<uhlc::HLC>,
17    /// The parsed and verified schema
18    pub schema: Arc<Schema>,
19    /// The unique actor ID for this database connection, distinguishing it from
20    /// other actors that gossip DB changes
21    pub actor_id: ActorId,
22}
23
24impl InitializedDb {
25    /// Attempts to initialize a CRSQL database with the specified schema, creating
26    /// it if it doesn't exist
27    pub async fn setup(db_path: &crate::Path, schema: &str) -> eyre::Result<Self> {
28        let partial_schema = corro_types::schema::parse_sql(schema)?;
29
30        let actor_id = {
31            // we need to set auto_vacuum before any tables are created
32            let db_conn = rusqlite::Connection::open(db_path)?;
33            db_conn.execute_batch("PRAGMA auto_vacuum = INCREMENTAL")?;
34
35            let conn = CrConn::init(db_conn)?;
36            conn.query_row("SELECT crsql_site_id();", [], |row| {
37                row.get::<_, ActorId>(0)
38            })?
39        };
40
41        let write_sema = Arc::new(tokio::sync::Semaphore::new(1));
42        let pool = SplitPool::create(&db_path, write_sema.clone()).await?;
43
44        let clock = Arc::new(
45            uhlc::HLCBuilder::default()
46                .with_id(actor_id.try_into().unwrap())
47                .with_max_delta(std::time::Duration::from_millis(300))
48                .build(),
49        );
50
51        let schema = {
52            let mut conn = pool.write_priority().await?;
53
54            let old_schema = {
55                corro_types::agent::migrate(clock.clone(), &mut conn)?;
56                let mut schema = corro_types::schema::init_schema(&conn)?;
57                schema.constrain()?;
58
59                schema
60            };
61
62            tokio::task::block_in_place(|| update_schema(&mut conn, old_schema, partial_schema))?
63        };
64
65        Ok(Self {
66            pool,
67            clock,
68            schema: Arc::new(schema),
69            actor_id,
70        })
71    }
72}

74/// We currently only support updating the schema at startup
75fn update_schema(
76    conn: &mut WriteConn,
77    old_schema: Schema,
78    new_schema: Schema,
79) -> eyre::Result<Schema> {
80    // clone the previous schema and apply
81    let mut new_schema = {
82        let mut schema = old_schema.clone();
83        for (name, def) in new_schema.tables.iter() {
84            // overwrite table because users are expected to return a full table def
85            schema.tables.insert(name.clone(), def.clone());
86        }
87        schema
88    };
89
90    new_schema.constrain()?;
91
92    let tx = conn.immediate_transaction()?;
93
94    corro_types::schema::apply_schema(&tx, &old_schema, &mut new_schema)?;
95
96    for tbl_name in new_schema.tables.keys() {
97        tx.execute("DELETE FROM __corro_schema WHERE tbl_name = ?", [tbl_name])?;
98
99        let n = tx.execute("INSERT INTO __corro_schema SELECT tbl_name, type, name, sql, 'api' AS source FROM sqlite_schema WHERE tbl_name = ? AND type IN ('table', 'index') AND name IS NOT NULL AND sql IS NOT NULL", [tbl_name])?;
100        tracing::info!("Updated {n} rows in __corro_schema for table {tbl_name}");
101    }
102
103    tx.commit()?;
104    Ok(new_schema)
105}

109/// Spawns an async task that periodically prints out locks that have surpassed
110/// the desired limits
111#[inline]
112pub fn spawn_lock_contention_printer(
113    registry: crate::types::agent::LockRegistry,
114    warn_threshold: Duration,
115    error_threshold: Duration,
116    check_interval: Duration,
117) -> tokio::task::JoinHandle<()> {
118    assert!(error_threshold > warn_threshold);
119
120    tokio::spawn(async move {
121        let mut interval = tokio::time::interval(check_interval);
122
123        let mut top = indexmap::IndexMap::<usize, corro_types::agent::LockMeta>::default();
124        loop {
125            interval.tick().await;
126
127            top.extend(
128                registry
129                    .map
130                    .read()
131                    .iter()
132                    .filter(|(_, meta)| meta.started_at.elapsed() >= warn_threshold)
133                    .take(10) // this is an ordered map, so taking the first few is gonna be the highest values
134                    .map(|(k, v)| (*k, v.clone())),
135            );
136
137            if top.is_empty() {
138                continue;
139            }
140
141            tracing::warn!(
142                held_locks = top.len(),
143                "lock registry shows locks held for a long time!"
144            );
145
146            for (id, lock) in top.drain(..) {
147                let duration = lock.started_at.elapsed();
148
149                if matches!(lock.state, corro_types::agent::LockState::Locked)
150                    && duration >= error_threshold
151                {
152                    tracing::error!(
153                        label = lock.label, id, kind = ?lock.kind, state = ?lock.state, ?duration, "lock exceeded error threshold"
154                    );
155                } else {
156                    tracing::warn!(
157                        label = lock.label, id, kind = ?lock.kind, state = ?lock.state, ?duration, "lock exceeded warning threshold"
158                    );
159                }
160            }
161        }
162    })
