3/// Error codes that can be sent as the close/reset for an HTTP/3 stream
4///
5/// These are just integers, so they are just a subset of HTTP status codes
6#[repr(u16)]
7pub enum ErrorCode {
8    Unknown = 0,
9    /// Success
10    Ok = 200,
11    /// The client request was malformed
12    BadRequest = 400,
13    /// There was an error deserializing or otherwise handling a handshake
14    BadHandshake = 402,
15    /// Could not find a requested resource (eg table)
16    NotFound = 404,
17    /// A length prefixed piece frame could not be read because the length could
18    /// not be read, or the frame could not be read before the end of the stream
19    LengthRequired = 411,
20    /// The size of a frame was too large
21    PayloadTooLarge = 413,
22    /// The size of a frame was too small
23    PayloadInsufficient = 414,
24    /// The client closed/aborted the connection before the server could send a
25    /// response
26    ClientClosed = 499,
27    /// Internal server error
28    InternalServerError = 500,
29    /// The server does not support this, ie, not implemented
30    ///
31    /// Corrosion does not support all possible SQL queries, but this is distinct
32    /// from client errors for malformed queries
33    Unsupported = 501,
34    /// The version of the client is not supported by the server
35    VersionNotSupported = 505,
36    /// The server send a bad response to a request
37    BadResponse = 588,
38}
39
40impl fmt::Display for ErrorCode {
41    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
42        match self {
43            Self::Unknown => f.write_str("0: unknown"),
44            Self::Ok => f.write_str("200: ok"),
45            Self::BadRequest => f.write_str("400: bad request"),
46            Self::BadHandshake => f.write_str("402: bad handshake"),
47            Self::NotFound => f.write_str("404: not found"),
48            Self::LengthRequired => f.write_str("411: length required"),
49            Self::PayloadTooLarge => f.write_str("413: payload too large"),
50            Self::PayloadInsufficient => f.write_str("414: payload insufficient"),
51            Self::ClientClosed => f.write_str("499: client closed"),
52            Self::InternalServerError => f.write_str("500: internal server error"),
53            Self::Unsupported => f.write_str("501: not supported"),
54            Self::VersionNotSupported => f.write_str("505: version not supported"),
55            Self::BadResponse => f.write_str("588: bad response"),
56        }
57    }
58}
59
60impl fmt::Debug for ErrorCode {
61    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
62        write!(f, "{self}")
63    }
64}
65
66impl From<ErrorCode> for quinn::VarInt {
67    fn from(value: ErrorCode) -> Self {
68        Self::from_u32(value as u32)
69    }
70}
71
72impl From<quinn::VarInt> for ErrorCode {
73    fn from(value: quinn::VarInt) -> Self {
74        match value.into_inner() {
75            200 => Self::Ok,
76            402 => Self::BadHandshake,
77            404 => Self::NotFound,
78            411 => Self::LengthRequired,
79            413 => Self::PayloadTooLarge,
80            414 => Self::PayloadInsufficient,
81            499 => Self::ClientClosed,
82            500 => Self::InternalServerError,
83            501 => Self::Unsupported,
84            505 => Self::VersionNotSupported,
85            588 => Self::BadResponse,
86            _ => Self::Unknown,
87        }
88    }
89}
