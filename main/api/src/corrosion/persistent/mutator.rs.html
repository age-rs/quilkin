26/// A DB mutator that will broadcast changes to any subscribers when a mutation
27/// occurs that matches a subscriber's query
28#[derive(Clone)]
29pub struct BroadcastingTransactor {
30    pool: SplitPool,
31    book: Booked,
32    subs: SubsManager,
33    tx: Option<BroadcastTx>,
34    clock: Arc<uhlc::HLC>,
35    id: ActorId,
36}
37
38impl BroadcastingTransactor {
39    pub async fn new(
40        id: ActorId,
41        clock: Arc<uhlc::HLC>,
42        pool: SplitPool,
43        subs: SubsManager,
44        lock_registry: LockRegistry,
45        tx: Option<BroadcastTx>,
46    ) -> Self {
47        let book = Booked::new(BookedVersions::new(id), lock_registry);
48
49        tokio::spawn({
50            let pool = pool.clone();
51            // acquiring the lock here means everything will have to wait for it to be ready
52            let mut booked = book.write_owned::<&str, _>("init", None).await;
53            async move {
54                let conn = pool.read().await?;
55                **booked = tokio::task::block_in_place(|| BookedVersions::from_conn(&conn, id))
56                    .expect("loading BookedVersions from db failed");
57                Ok::<_, eyre::Report>(())
58            }
59        });
60
61        Self {
62            pool,
63            book,
64            clock,
65            subs,
66            id,
67            tx,
68        }
69    }
70
71    #[inline]
72    pub fn actor_id(&self) -> ActorId {
73        self.id
74    }
75
76    pub async fn make_broadcastable_changes<F, T>(
77        &self,
78        timeout: Option<Duration>,
79        f: F,
80    ) -> Result<(T, Option<CrsqlDbVersion>, Duration), ChangeError>
81    where
82        F: FnOnce(&InterruptibleTransaction<Transaction<'_>>) -> Result<T, ChangeError>,
83    {
84        let mut conn = self.pool.write_priority().await?;
85
86        let mut book_writer = self
87            .book
88            .write::<&str, _>("make_broadcastable_changes(booked writer)", None)
89            .await;
90
91        let actor_id = self.id;
92        let start = Instant::now();
93        let clock = self.clock.clone();
94        let ts = broadcast::Timestamp::from(clock.new_timestamp());
95
96        tokio::task::block_in_place(move || {
97            let tx = conn
98                .immediate_transaction()
99                .map_err(|source| ChangeError::Rusqlite {
100                    source,
101                    actor_id: Some(actor_id),
102                    version: None,
103                })?;
104
105            let tx = InterruptibleTransaction::new(tx, timeout, "local_changes");
106
107            let _unused = tx
108                .prepare_cached("SELECT crsql_set_ts(?)")
109                .map_err(|source| ChangeError::Rusqlite {
110                    source,
111                    actor_id: Some(actor_id),
112                    version: None,
113                })?
114                .query_row([&ts], |row| row.get::<_, String>(0))
115                .map_err(|source| ChangeError::Rusqlite {
116                    source,
117                    actor_id: Some(actor_id),
118                    version: None,
119                })?;
120
121            // Execute whatever might mutate state data
122            let ret = f(&tx)?;
123
124            let insert_info = insert_local_changes(actor_id, &clock, &tx, &mut book_writer)?;
125            tx.commit().map_err(|source| ChangeError::Rusqlite {
126                source,
127                actor_id: Some(actor_id),
128                version: insert_info.as_ref().map(|info| info.db_version),
129            })?;
130
131            let elapsed = start.elapsed();
132
133            match insert_info {
134                None => Ok((ret, None, elapsed)),
135                Some(change::InsertChangesInfo {
136                    db_version,
137                    last_seq,
138                    ts,
139                    snap,
140                }) => {
141                    tracing::debug!(%db_version, ?last_seq, "committed tx");
142
143                    book_writer.commit_snapshot(snap);
144
145                    let pool = self.pool.clone();
146                    let subs = self.subs.clone();
147                    let tx = self.tx.clone();
148
149                    spawn::spawn_counted(async move {
150                        broadcast_changes(&pool, actor_id, &subs, tx, db_version, last_seq, ts)
151                            .await
152                    });
153
154                    Ok::<_, ChangeError>((ret, Some(db_version), elapsed))
155                }
156            }
157        })
158    }
159}

161#[async_trait::async_trait]
162impl super::server::DbMutator for BroadcastingTransactor {
163    async fn connected(&self, peer: Peer, icao: IcaoCode, qcmp_port: u16) {
164        let mut dc = smallvec::SmallVec::<[_; 1]>::new();
165        {
166            let mut dc = db::write::Datacenter(&mut dc);
167            dc.insert(peer, qcmp_port, icao);
168        }
169
170        let res = self
171            .make_broadcastable_changes(None, move |tx| {
172                db::write::exec_interruptible(tx, dc).map_err(|source| ChangeError::Rusqlite {
173                    source,
174                    actor_id: Some(self.id),
175                    version: None,
176                })
177            })
178            .await;
179
180        match res {
181            Ok((_rows, _version, elapsed)) => {
182                tracing::debug!(%peer, ?elapsed, "peer added");
183            }
184            Err(error) => {
185                tracing::error!(%peer, %error, "failed to insert dc");
186            }
187        }
188    }
189
190    async fn execute(&self, peer: Peer, statements: &[p::ServerChange]) -> ExecResponse {
191        let start = std::time::Instant::now();
192
193        let mut v = smallvec::SmallVec::<[_; 32]>::new();
194        {
195            for s in statements {
196                match s {
197                    p::ServerChange::Upsert(i) => {
198                        let mut srv = db::write::Server::for_peer(peer, &mut v);
199                        for i in i {
200                            srv.upsert(&i.endpoint, i.icao, &i.tokens);
201                        }
202                    }
203                    p::ServerChange::Remove(r) => {
204                        let mut srv = db::write::Server::for_peer(peer, &mut v);
205                        for r in r {
206                            srv.remove_immediate(r);
207                        }
208                    }
209                    p::ServerChange::Update(u) => {
210                        let mut srv = db::write::Server::for_peer(peer, &mut v);
211                        for u in u {
212                            srv.update(&u.endpoint, u.icao, u.tokens.as_ref());
213                        }
214                    }
215                    p::ServerChange::UpdateMutator(mu) => {
216                        let mut dc = db::write::Datacenter(&mut v);
217                        dc.update(peer, mu.qcmp_port, mu.icao);
218                    }
219                }
220            }
221        }
222
223        let mut results = Vec::with_capacity(statements.len());
224
225        let res = self
226            .make_broadcastable_changes(None, |tx| {
227                let mut rows = 0;
228                for statement in v {
229                    match db::write::exec_single_interruptible(tx, statement) {
230                        Ok(rows_affected) => {
231                            rows += rows_affected;
232                            results.push(ExecResult::Execute {
233                                rows_affected,
234                                time: start.elapsed().as_secs_f64(),
235                            });
236                        }
237                        Err(error) => {
238                            results.push(ExecResult::Error {
239                                error: error.to_string(),
240                            });
241                        }
242                    }
243                }
244
245                Ok(rows)
246            })
247            .await;
248
249        let version = match res {
250            Ok((rows_affected, version, elapsed)) => {
251                tracing::debug!(%peer, ?elapsed, rows_affected, "updated servers");
252                version.map(u64::from)
253            }
254            Err(error) => {
255                tracing::error!(%peer, %error, "failed to update servers");
256                results.push(ExecResult::Error {
257                    error: error.to_string(),
258                });
259                None
260            }
261        };
262
263        ExecResponse {
264            results,
265            time: start.elapsed().as_secs_f64(),
266            version,
267            actor_id: Some(self.id.to_string()),
268        }
269    }
270
271    async fn disconnected(&self, peer: Peer) {
272        let mut dc = smallvec::SmallVec::<[_; 2]>::new();
273        {
274            let mut dc = db::write::Datacenter(&mut dc);
275            dc.remove(peer, None);
276        }
277
278        let res = self
279            .make_broadcastable_changes(None, move |tx| {
280                db::write::exec_interruptible(tx, dc).map_err(|source| ChangeError::Rusqlite {
281                    source,
282                    actor_id: Some(self.id),
283                    version: None,
284                })
285            })
286            .await;
287
288        match res {
289            Ok((_rows, _version, elapsed)) => {
290                tracing::debug!(%peer, ?elapsed, "peer removed");
291            }
292            Err(error) => {
293                tracing::error!(%peer, %error, "failed to remove dc");
294            }
295        }
296    }
297}

299pub fn insert_local_changes(
300    actor_id: ActorId,
301    clock: &uhlc::HLC,
302    tx: &rusqlite::Connection,
303    book_writer: &mut tokio::sync::RwLockWriteGuard<'_, BookedVersions>,
304) -> Result<Option<change::InsertChangesInfo>, ChangeError> {
305    let db_version: CrsqlDbVersion = tx
306        .prepare_cached("SELECT crsql_peek_next_db_version()")
307        .map_err(|source| ChangeError::Rusqlite {
308            source,
309            actor_id: Some(actor_id),
310            version: None,
311        })?
312        .query_row((), |row| row.get(0))
313        .map_err(|source| ChangeError::Rusqlite {
314            source,
315            actor_id: Some(actor_id),
316            version: None,
317        })?;
318
319    let version_info: (Option<CrsqlSeq>, Option<broadcast::Timestamp>) = tx
320        .prepare_cached(
321            "SELECT MAX(seq), MAX(ts) FROM crsql_changes WHERE site_id = ? AND db_version = ?;",
322        )
323        .map_err(|source| ChangeError::Rusqlite {
324            source,
325            actor_id: Some(actor_id),
326            version: None,
327        })?
328        .query_row((actor_id, db_version), |row| Ok((row.get(0)?, row.get(1)?)))
329        .map_err(|source| ChangeError::Rusqlite {
330            source,
331            actor_id: Some(actor_id),
332            version: None,
333        })?;
334
335    match version_info {
336        (None, None) => Ok(None),
337        (None, Some(ts)) => {
338            tracing::warn!(%db_version, ?ts, "found db_version without seq");
339            Ok(None)
340        }
341        (Some(last_seq), ts) => {
342            let ts = ts.unwrap_or_else(|| {
343                tracing::warn!(%db_version, ?ts, "found db_version without seq");
344                broadcast::Timestamp::from(clock.new_timestamp())
345            });
346
347            tracing::debug!(%db_version, %last_seq, ?ts, "found db_version");
348
349            let db_versions = db_version..=db_version;
350
351            let mut snap = book_writer.snapshot();
352            snap.insert_db(tx, [db_versions].into())
353                .map_err(|source| ChangeError::Rusqlite {
354                    source,
355                    actor_id: Some(actor_id),
356                    version: Some(db_version),
357                })?;
358
359            Ok(Some(change::InsertChangesInfo {
360                db_version,
361                last_seq,
362                ts,
363                snap,
364            }))
365        }
366    }
367}

369pub async fn broadcast_changes(
370    pool: &SplitPool,
371    actor_id: ActorId,
372    subs: &SubsManager,
373    tx: Option<BroadcastTx>,
374    db_version: CrsqlDbVersion,
375    last_seq: CrsqlSeq,
376    ts: broadcast::Timestamp,
377) -> Result<(), broadcast::BroadcastError> {
378    let conn = pool.read().await?;
379
380    tokio::task::block_in_place(|| {
381        let mut prepped = conn.prepare_cached(
382            r#"
383                SELECT "table", pk, cid, val, col_version, db_version, seq, site_id, cl
384                    FROM crsql_changes
385                    WHERE db_version = ?
386                    AND site_id = crsql_site_id()
387                    ORDER BY seq ASC
388            "#,
389        )?;
390        let rows = prepped.query_map([db_version], change::row_to_change)?;
391        let chunked =
392            change::ChunkedChanges::new(rows, CrsqlSeq(0), last_seq, change::MAX_CHANGES_BYTE_SIZE);
393        for changes_seqs in chunked {
394            match changes_seqs {
395                Ok((changes, seqs)) => {
396                    tracing::trace!(num_changes = changes.len(), ?seqs, "broadcasting changes");
397
398                    let changeset = corro_types::broadcast::Changeset::FullV2 {
399                        actor_id,
400                        version: db_version,
401                        changes,
402                        seqs,
403                        last_seq,
404                        ts,
405                    };
406                    match_changes(subs, &changeset, db_version);
407
408                    if let Some(tx) = tx.clone() {
409                        tokio::spawn(async move {
410                            if tx
411                                .send(broadcast::BroadcastInput::AddBroadcast(
412                                    broadcast::BroadcastV1::Change(broadcast::ChangeV1 {
413                                        actor_id,
414                                        changeset,
415                                    }),
416                                ))
417                                .await
418                                .is_err()
419                            {
420                                tracing::error!("could not send change message for broadcast");
421                            }
422                        });
423                    }
424                }
425                Err(error) => {
426                    tracing::error!(
427                        %db_version, %error, "could not process crsql change"
428                    );
429                    break;
430                }
431            }
432        }
433        Ok::<_, rusqlite::Error>(())
434    })?;
435
436    Ok(())
