19#[derive(thiserror::Error, Debug)]
20pub enum StreamError {
21    #[error(transparent)]
22    Connect(#[from] quinn::ConnectionError),
23    #[error(transparent)]
24    Write(#[from] quinn::WriteError),
25    #[error(transparent)]
26    Read(#[from] quinn::ReadError),
27    #[error(transparent)]
28    ReadExact(#[from] quinn::ReadExactError),
29    #[error(transparent)]
30    Reset(#[from] quinn::ResetError),
31    #[error(transparent)]
32    Json(serde_json::Error),
33    #[error(
34        "expected a chunk of JSON length {} but received {}",
35        expected,
36        received
37    )]
38    LengthMismatch { expected: usize, received: usize },
39    #[error("stream ended")]
40    StreamEnded,
41}
42
43use codec::LengthReadError as Lre;
44
45impl From<Lre> for StreamError {
46    fn from(value: Lre) -> Self {
47        match value {
48            Lre::Json(json) => Self::Json(json),
49            Lre::LengthMismatch { expected, received } => {
50                Self::LengthMismatch { expected, received }
51            }
52            Lre::ReadExact(re) => Self::ReadExact(re),
53            Lre::Read(r) => Self::Read(r),
54            Lre::StreamEnded => Self::StreamEnded,
55        }
56    }
57}

94    pub async fn connect_insecure(addr: SocketAddr) -> Result<Self, ConnectError> {
95        let ep = quinn::Endpoint::client((std::net::Ipv6Addr::LOCALHOST, 0).into())?;
96
97        let conn = ep
98            .connect_with(
99                quinn_plaintext::client_config(),
100                addr,
101                &addr.ip().to_string(),
102            )?
103            .await?;
104
105        // This is really infallible
106        let local_addr = ep.local_addr()?;
107
108        Ok(Self { conn, local_addr })
109    }

118        self.conn.remote_address()
119    }

121    /// Closes the connection to the upstream server
122    #[inline]
123    pub async fn shutdown(self) {
124        drop(self.conn);

134impl MutationClient {
135    pub async fn connect(
136        inner: Client,
137        qcmp_port: u16,
138        icao: IcaoCode,
139    ) -> Result<Self, ConnectError> {
140        let (mut send, mut recv) = inner.conn.open_bi().await?;
141
142        // We need to actually send something for the connection to be fully established
143        let req_buf =
144            proto::v1::Request::Mutate(proto::v1::MutateRequest { qcmp_port, icao }).write()?;
145
146        send.write_chunk(req_buf).await.map_err(StreamError::from)?;
147
148        let res = codec::read_length_prefixed(&mut recv)
149            .await
150            .map_err(StreamError::from)?;
151
152        let vb = proto::VersionedBuf::try_parse(&res)?;
153        let peer_version = vb.version;
154        let response = vb.deserialize_response()?;
155
156        match response {
157            proto::Response::V1(res) => {
158                use proto::v1;
159
160                match res {
161                    v1::Response::Ok(v1::OkResponse::Mutate(_)) => {}
162                    v1::Response::Ok(invalid) => {
163                        tracing::error!(
164                            ?invalid,
165                            peer_version,
166                            "received a non-mutate response to a mutate request"
167                        );
168                        return Err(proto::Error::InvalidResponse.into());
169                    }
170                    v1::Response::Err { code, message } => {
171                        return Err(proto::Error::ErrorResponse { code, message }.into());
172                    }
173                }
174            }
175        }
176
177        let (tx, mut reqrx) = mpsc::unbounded_channel();
178
179        let task = tokio::task::spawn(async move {
180            let mut func = async || -> Result<Option<quinn::VarInt>, StreamError> {
181                match peer_version {
182                    1 => loop {
183                        let (msg, comp): (_, ResponseTx) = tokio::select! {
184                            res = recv.received_reset() => {
185                                return res.map_err(StreamError::Reset);
186                            }
187                            req = reqrx.recv() => {
188                                let Some(req) = req else {
189                                    // Initiate a shutdown, the server will see we've reset its receiver
190                                    // and it will exit the loop when it has finished all messages, closing
191                                    // its sender, which we will know when we receive a reset (or it times out)
192                                    let _ = send.reset(ErrorCode::Ok.into());
193                                    let _ = send.finish();
194
195                                    tracing::debug!("waiting for server to shutdown this stream...");
196                                    drop(recv.received_reset().await);
197                                    tracing::debug!("client finished");
198                                    break;
199                                };
200
201                                req
202                            }
203                        };
204
205                        send.write_chunk(msg).await?;
206                        let res = codec::read_length_prefixed_jsonb::<ExecResponse>(&mut recv)
207                            .await
208                            .map_err(StreamError::from);
209
210                        if let Err(error) = &res {
211                            tracing::error!(%error, "error occurred reading response to transaction");
212                        }
213
214                        if comp.send(res).is_err() {
215                            tracing::warn!("transaction response could not be sent to queuer");
216                        }
217                    },
218                    _invalid => {
219                        return Err(StreamError::Connect(
220                            quinn::ConnectionError::VersionMismatch,
221                        ));
222                    }
223                }
224
225                Ok(None)
226            };
227
228            func().await
229        });
230
231        Ok(Self { inner, tx, task })
232    }

234    #[inline]
235    pub async fn update(
236        &self,
237        qcmp_port: Option<u16>,
238        icao: Option<IcaoCode>,
239    ) -> Result<ExecResponse, TransactionError> {
240        if qcmp_port.is_none() && icao.is_none() {
241            return Ok(ExecResponse {
242                results: vec![ExecResult::Error {
243                    error: "neither the QCMP port nor ICAO were provided".into(),
244                }],
245                time: 0.,
246                version: None,
247                actor_id: None,
248            });
249        }
250
251        self.transactions(&[proto::v1::ServerChange::UpdateMutator(
252            proto::v1::MutatorUpdate { qcmp_port, icao },
253        )])
254        .await
255    }
256
257    #[inline]
258    pub async fn transactions(
259        &self,
260        change: &[proto::v1::ServerChange],
261    ) -> Result<ExecResponse, TransactionError> {
262        let buf = codec::write_length_prefixed_jsonb(&change)?;
263
264        let (tx, rx) = oneshot::channel();
265        self.tx
266            .send((buf.freeze(), tx))
267            .map_err(|_e| TransactionError::TaskShutdown)?;
268        Ok(rx.await.map_err(|_e| TransactionError::TaskShutdown)??)
269    }
270
271    pub async fn shutdown(self) {
272        drop(self.tx);
273        if let Ok(Err(error)) = self.task.await {
274            tracing::warn!(%error, "stream exited with error");
275        }
276        self.inner.shutdown().await;
277    }

288pub struct SubClientStream {
289    /// The unique identifier for this subscription
290    pub id: Uuid,
291    /// The hash of the query that was subscribed to
292    pub query_hash: String,
293    /// The stream of changes that match the subscription query
294    ///
295    /// Dropping this will terminate the subscription on the remote server,
296    /// but ideally one would call [`SubscriptionClient::shutdown`] to gracefully
297    /// close the stream
298    pub rx: mpsc::UnboundedReceiver<QueryEvent>,
299}
300
301impl SubscriptionClient {
302    pub async fn connect(
303        inner: Client,
304        params: crate::pubsub::SubParamsv1,
305    ) -> Result<(Self, SubClientStream), ConnectError> {
306        let (mut send, mut recv) = inner.conn.open_bi().await?;
307
308        let res = Self::handshake(&mut send, &mut recv, params).await?;
309
310        let (tx, reqrx) = mpsc::unbounded_channel();
311        let (stx, srx) = oneshot::channel();
312
313        let sub_id = res.id;
314
315        let scs = SubClientStream {
316            id: res.id,
317            query_hash: res.query_hash,
318            rx: reqrx,
319        };
320
321        let task = tokio::task::spawn(async move {
322            Self::run_subscription_loop(recv, send, tx, srx)
323                .instrument(tracing::info_span!("subscription", %sub_id))
324                .await
325        });
326
327        Ok((
328            Self {
329                inner,
330                tx: stx,
331                task,
332            },
333            scs,
334        ))
335    }
336
337    async fn run_subscription_loop(
338        mut recv: quinn::RecvStream,
339        send: quinn::SendStream,
340        tx: mpsc::UnboundedSender<QueryEvent>,
341        mut srx: oneshot::Receiver<ErrorCode>,
342    ) -> Result<Option<quinn::VarInt>, StreamError> {
343        tracing::info!("started subscription stream");
344
345        'io: loop {
346            tokio::select! {
347                res = codec::read_length_prefixed(&mut recv) => {
348                    match res {
349                        Ok(buf) => {
350                            for item in pubsub::SubscriptionStream::new(buf) {
351                                match item {
352                                    Ok(item) => {
353                                        if tx.send(item).is_err() {
354                                            tracing::info!("subscription stream receiver closed, closing stream");
355                                            let _ = recv.stop(ErrorCode::Unknown.into());
356                                            break 'io;
357                                        }
358                                    }
359                                    Err(error) => {
360                                        tracing::error!(%error, "failed to deserialize event from subscription stream");
361                                    }
362                                }
363                            }
364                        }
365                        Err(_err) => {
366                            match recv.received_reset().await {
367                                Ok(code) => {
368                                    tracing::warn!(code = ?code.map(ErrorCode::from), "server shutdown subscription stream");
369                                }
370                                Err(error) => {
371                                    tracing::warn!(%error, "server shutdown subscription stream abnormally");
372                                }
373                            }
374
375                            break;
376                        }
377                    }
378                }
379                code = &mut srx => {
380                    let code = code.unwrap_or_else(|_| {
381                        tracing::error!("SubscriptionClient dropped without calling shutdown");
382                            ErrorCode::Unknown
383                    });
384                    drop(recv.stop(code.into()));
385                    break;
386                }
387            }
388        }
389
390        tracing::info!("waiting for remote shutdown");
391        let stime = std::time::Instant::now();
392        let res = send.stopped().await;
393        tracing::info!(time = ?stime.elapsed(), result = ?res, "remote shutdown complete");
394
395        Ok(None)
396    }
397
398    async fn handshake(
399        send: &mut quinn::SendStream,
400        recv: &mut quinn::RecvStream,
401        params: crate::pubsub::SubParamsv1,
402    ) -> Result<proto::v1::SubscribeResponse, ConnectError> {
403        // We need to actually send something for the connection to be fully established
404        let req_buf = proto::v1::Request::Subscribe(proto::v1::SubscribeRequest(params)).write()?;
405
406        send.write_chunk(req_buf).await.map_err(StreamError::from)?;
407
408        let res = codec::read_length_prefixed(recv)
409            .await
410            .map_err(StreamError::from)?;
411
412        macro_rules! bad_response {
413            ($op:expr) => {
414                match $op {
415                    Ok(r) => r,
416                    Err(error) => {
417                        let _ = recv.stop(ErrorCode::BadResponse.into());
418                        return Err(error.into());
419                    }
420                }
421            };
422        }
423
424        let vb = bad_response!(proto::VersionedBuf::try_parse(&res));
425        let peer_version = vb.version;
426        let response = bad_response!(vb.deserialize_response());
427
428        use proto::v1;
429
430        match response {
431            proto::Response::V1(v1::Response::Ok(v1::OkResponse::Subscribe(sr))) => Ok(sr),
432            proto::Response::V1(v1::Response::Ok(invalid)) => {
433                tracing::error!(
434                    ?invalid,
435                    peer_version,
436                    "received a non-subscribe response to a subscribe request"
437                );
438                Err(proto::Error::InvalidResponse.into())
439            }
440            proto::Response::V1(v1::Response::Err { code, message }) => {
441                Err(proto::Error::ErrorResponse { code, message }.into())
442            }
443        }
444    }
445
446    /// Shuts down this client, sending the specified code to the server if it
447    /// is able
448    pub async fn shutdown(self, code: ErrorCode) {
449        let _ = self.tx.send(code);
450        if let Ok(Err(error)) = self.task.await {
451            tracing::warn!(%error, "stream exited with error");
452        }
453        self.inner.shutdown().await;
454    }
