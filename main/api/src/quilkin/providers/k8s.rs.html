36fn track_event<T>(kind: &'static str, event: Event<T>) -> Event<T> {
37    let ty = match &event {
38        Event::Apply(_) => "apply",
39        Event::Init => "init",
40        Event::InitApply(_) => "init-apply",
41        Event::InitDone => "init-done",
42        Event::Delete(_) => "done",
43    };
44
45    metrics::k8s::events_total(kind, ty).inc();
46    event
47}

49pub(crate) async fn update_leader_lock(
50    client: kube::Client,
51    namespace: impl AsRef<str>,
52    lease_name: impl Into<String>,
53    holder_id: impl Into<String>,
54    leader_lock: config::LeaderLock,
55) -> crate::Result<()> {
56    let leadership = LeaseLock::new(
57        client,
58        namespace.as_ref(),
59        LeaseLockParams {
60            holder_id: holder_id.into(),
61            lease_name: lease_name.into(),
62            lease_ttl: std::time::Duration::from_secs(1),
63        },
64    );
65
66    loop {
67        match leadership.try_acquire_or_renew().await {
68            Ok(ll) => leader_lock.store(ll.acquired_lease),
69            Err(error) => tracing::warn!(%error),
70        }
71
72        tokio::time::sleep(std::time::Duration::from_millis(500)).await;
73    }
74}
75pub fn update_filters_from_configmap(
76    client: kube::Client,
77    namespace: impl AsRef<str>,
78    filters: config::filter::FilterChainConfig,
79) -> impl Stream<Item = crate::Result<(), eyre::Error>> {
80    async_stream::stream! {
81        let mut cmap = None;
82        for await event in configmap_events(client, namespace) {
83            tracing::trace!("new configmap event");
84
85            let event = match event {
86                Ok(event) => event,
87                Err(error) => {
88                    metrics::k8s::errors_total(CONFIGMAP, &error).inc();
89                    yield Err(error.into());
90                    continue;
91                }
92            };
93
94            let configmap = match track_event(CONFIGMAP, event) {
95                Event::Apply(configmap) => configmap,
96                Event::Init => { yield Ok(()); continue; }
97                Event::InitApply(configmap) => {
98                    if cmap.is_none() {
99                        cmap = Some(configmap);
100                    }
101                    yield Ok(());
102                    continue;
103                }
104                Event::InitDone => {
105                    if let Some(cmap) = cmap.take() {
106                        cmap
107                    } else {
108                        yield Ok(());
109                        continue;
110                    }
111                }
112                Event::Delete(_) => {
113                    metrics::k8s::filters(false);
114                    filters.store(Default::default());
115                    yield Ok(());
116                    continue;
117                }
118            };
119
120            let data = configmap.data.ok_or_else(|| eyre::eyre!("configmap data missing"))?;
121            let data = data.get("quilkin.yaml").ok_or_else(|| eyre::eyre!("quilkin.yaml property not found"))?;
122            let data: serde_json::Map<String, serde_json::Value> = serde_yaml::from_str(data)?;
123
124            if let Some(de_filters) = data
125                .get("filters")
126                    .cloned()
127                    .map(serde_json::from_value)
128                    .transpose()?
129            {
130                metrics::k8s::filters(true);
131                filters.store(de_filters);
132            }
133
134            yield Ok(());
135        }
136    }
137}

139fn configmap_events(
140    client: kube::Client,
141    namespace: impl AsRef<str>,
142) -> impl Stream<Item = Result<Event<ConfigMap>, kube::runtime::watcher::Error>> {
143    let config_namespace = namespace.as_ref();
144    let configmap: kube::Api<ConfigMap> = kube::Api::namespaced(client, config_namespace);
145    let config_writer = kube::runtime::reflector::store::Writer::<ConfigMap>::default();
146    let configmap_stream = kube::runtime::watcher(
147        configmap,
148        kube::runtime::watcher::Config::default().labels("quilkin.dev/configmap=true"),
149    );
150    kube::runtime::reflector(config_writer, configmap_stream)
151}

153fn gameserver_events(
154    client: kube::Client,
155    namespace: impl AsRef<str>,
156) -> impl Stream<Item = Result<Event<DeserializeGuard<GameServer>>, kube::runtime::watcher::Error>>
157{
158    let gameservers_namespace = namespace.as_ref();
159    let gameservers: kube::Api<DeserializeGuard<GameServer>> =
160        kube::Api::namespaced(client, gameservers_namespace);
161    let gs_writer =
162        kube::runtime::reflector::store::Writer::<DeserializeGuard<GameServer>>::default();
163    let mut config = kube::runtime::watcher::Config::default()
164        // Default timeout is 5 minutes, far too slow for us to react.
165        .timeout(15);
166
167    // Retreive unbounded results.
168    config.page_size = None;
169
170    let gameserver_stream = kube::runtime::watcher(gameservers, config);
171    kube::runtime::reflector(gs_writer, gameserver_stream)
172}

174fn validate_gameserver(
175    result: DeserializeGuard<GameServer>,
176    ads: Option<&crate::config::AddressSelector>,
177) -> Option<crate::net::endpoint::Endpoint> {
178    match result.0 {
179        Ok(server) => {
180            if server.is_allocated() {
181                if let Some(ep) = server.endpoint(ads) {
182                    tracing::trace!(endpoint=%ep.address, metadata=serde_json::to_string(&ep.metadata).unwrap(), "applying server");
183                    metrics::k8s::gameservers_total_valid();
184                    Some(ep)
185                } else {
186                    tracing::warn!(
187                        server = serde_json::to_string(&server).unwrap(),
188                        "skipping invalid server"
189                    );
190                    metrics::k8s::gameservers_total_invalid();
191                    None
192                }
193            } else {
194                tracing::debug!(
195                    server = serde_json::to_string(&server).unwrap(),
196                    "skipping unallocated server"
197                );
198                metrics::k8s::gameservers_total_unallocated();
199                None
200            }
201        }
202        Err(error) => {
203            tracing::debug!(error=%error.error, metadata=serde_json::to_string(&error.metadata).unwrap(), "couldn't decode gameserver event");
204            metrics::k8s::errors_total(GAMESERVER, &error);
205            None
206        }
207    }
208}

210pub fn update_endpoints_from_gameservers(
211    client: kube::Client,
212    namespace: impl AsRef<str>,
213    clusters: config::Watch<ClusterMap>,
214    locality: Option<Locality>,
215    address_selector: Option<crate::config::AddressSelector>,
216) -> impl Stream<Item = crate::Result<(), eyre::Error>> {
217    async_stream::stream! {
218        let mut servers = BTreeSet::new();
219
220        for await event in gameserver_events(client, namespace) {
221            let ads = address_selector.as_ref();
222            let event = match event {
223                Ok(event) => event,
224                Err(error) => {
225                    tracing::warn!(%error, "gameserver watch error");
226                    continue;
227                }
228            };
229
230            match track_event(GAMESERVER, event) {
231                Event::Apply(result) => {
232                    let span = tracing::trace_span!("k8s::gameservers::apply");
233                    let _enter = span.enter();
234                    let Some(endpoint) = validate_gameserver(result, ads) else {
235                        continue;
236                    };
237                    tracing::debug!(endpoint=%serde_json::to_value(&endpoint).unwrap(), "Adding endpoint");
238                    metrics::k8s::gameservers_total_valid();
239                    clusters.write().replace(None, locality.clone(), endpoint);
240                }
241                Event::Init => {},
242                Event::InitApply(result) => {
243                    let span = tracing::trace_span!("k8s::gameservers::init_apply");
244                    let _enter = span.enter();
245                    let Some(endpoint) = validate_gameserver(result, ads) else {
246                        continue;
247                    };
248
249                    tracing::trace!(%endpoint.address, endpoint.metadata=serde_json::to_string(&endpoint.metadata).unwrap(), "applying server");
250                    metrics::k8s::gameservers_total_valid();
251                    servers.insert(endpoint);
252                }
253                Event::InitDone => {
254                    let span = tracing::trace_span!("k8s::gameservers::init_done");
255                    let _enter = span.enter();
256                    tracing::debug!("received restart event from k8s");
257
258                    tracing::trace!(
259                        endpoints=%serde_json::to_value(servers.clone()).unwrap(),
260                        "Restarting with endpoints"
261                    );
262
263                    clusters.write().insert(None, locality.clone(), std::mem::take(&mut servers));
264                }
265                Event::Delete(result) => {
266                    let span = tracing::trace_span!("k8s::gameservers::delete");
267                    let _enter = span.enter();
268                    let server = match result.0 {
269                        Ok(server) => server,
270                        Err(error) => {
271                            metrics::k8s::errors_total(GAMESERVER, &error);
272                            tracing::debug!(%error, metadata=serde_json::to_string(&error.metadata).unwrap(), "couldn't decode gameserver event");
273                            continue;
274                        }
275                    };
276
277                    let found = if let Some(endpoint) = server.endpoint(ads) {
278                        tracing::debug!(%endpoint.address, endpoint.metadata=serde_json::to_string(&endpoint.metadata).unwrap(), "deleting by endpoint");
279                        clusters.write().remove_endpoint(&endpoint)
280                    } else {
281                        tracing::debug!(server.metadata.name=%server.metadata.name.clone().unwrap_or_default(), "deleting by server name");
282                        clusters.write().remove_endpoint_if(|endpoint| {
283                            endpoint.metadata.unknown.get("name") == server.metadata.name.clone().map(From::from).as_ref()
284                        })
285                    };
286
287                    metrics::k8s::gameservers_deletions_total(found);
288                    if !found {
289                        tracing::debug!(
290                            endpoint=%serde_json::to_value(server.endpoint(ads)).unwrap(),
291                            server.metadata.name=%serde_json::to_value(server.metadata.name).unwrap(),
292                            "received unknown gameserver to delete from k8s"
293                        );
294                    } else {
295                        tracing::debug!(
296                            endpoint=%serde_json::to_value(server.endpoint(ads)).unwrap(),
297                            server.metadata.name=%serde_json::to_value(server.metadata.name).unwrap(),
298                            "deleted gameserver"
299                        );
300                    }
301                }
302            };
303
304            crate::metrics::apply_clusters(&clusters);
305            yield Ok(());
306        }
307    }
