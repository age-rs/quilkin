26/// Filter that only allows packets to be passed to Endpoints that have a matching
27/// `connection_id` to the token stored in the Filter's dynamic metadata.
28#[derive(Default)]
29pub struct TokenRouter {
30    config: Config,
31}
32
33impl TokenRouter {
34    pub fn testing(config: Option<Config>) -> Self {
35        Self {
36            config: config.unwrap_or_default(),
37        }
38    }
39}

41impl StaticFilter for TokenRouter {
42    const NAME: &'static str = "quilkin.filters.token_router.v1alpha1.TokenRouter";
43    type Configuration = Config;
44    type BinaryConfiguration = proto::TokenRouter;
45
46    fn try_from_config(config: Option<Self::Configuration>) -> Result<Self, CreationError> {
47        Ok(Self {
48            config: config.unwrap_or_default(),
49        })
50    }
51}
52
53impl Filter for TokenRouter {
54    fn read<P: PacketMut>(&self, ctx: &mut ReadContext<'_, P>) -> Result<(), FilterError> {
55        match ctx.metadata.get(&self.config.metadata_key) {
56            Some(metadata::Value::Bytes(token)) => {
57                let tok = crate::net::cluster::Token::new(token);
58
59                ctx.endpoints.addresses_for_token(tok, ctx.destinations);
60
61                if ctx.destinations.is_empty() {
62                    Err(FilterError::TokenRouter(RouterError::NoEndpointMatch {
63                        token: token.clone(),
64                    }))
65                } else {
66                    Ok(())
67                }
68            }
69            Some(_value) => unreachable!(
70                "this means the capture filter has regressed, it only ever captures byte slices"
71            ),
72            None => Err(FilterError::TokenRouter(RouterError::NoTokenFound)),
73        }
74    }
75}

97pub enum RouterError {
98    NoTokenFound,
99    NoEndpointMatch { token: bytes::Bytes },
100}
101
102impl RouterError {
103    #[inline]
104    pub fn discriminant(&self) -> &'static str {
105        match self {
106            Self::NoEndpointMatch { .. } => "filter::token_router::no endpoint match",
107            Self::NoTokenFound => "filter::token_router::no token found",
108        }
109    }
110}
111
112use std::fmt;
113
114impl fmt::Display for RouterError {
115    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
116        match self {
117            Self::NoEndpointMatch { token } => {
118                write!(
119                    f,
120                    "no endpoint matched token `{}`",
121                    base64::display::Base64Display::new(
122                        token,
123                        &base64::engine::general_purpose::STANDARD
124                    )
125                )
126            }
127            Self::NoTokenFound => f.write_str("routing token not captured"),
128        }
129    }
130}
131
132impl fmt::Debug for RouterError {
133    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
134        match self {
135            Self::NoEndpointMatch { .. } => f.write_str("no endpoint matched routing token"),
136            Self::NoTokenFound => f.write_str("routing token not captured"),
137        }
138    }
139}
140
141impl Eq for RouterError {}
142
143impl PartialEq for RouterError {
144    fn eq(&self, other: &Self) -> bool {
145        match (self, other) {
146            (Self::NoTokenFound, Self::NoTokenFound) => true,
147            (Self::NoEndpointMatch { token: a }, Self::NoEndpointMatch { token: b }) => a == b,
148            _ => false,
149        }
150    }
151}
152
153impl std::hash::Hash for RouterError {
154    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
155        std::hash::Hash::hash(&std::mem::discriminant(self), state);
156
157        match self {
158            Self::NoEndpointMatch { token } => state.write(token),
159            Self::NoTokenFound => {}
160        }
161    }
162}

164#[derive(Serialize, Deserialize, Debug, Eq, PartialEq, schemars::JsonSchema)]
165#[serde(default)]
166pub struct Config {
167    /// the key to use when retrieving the token from the Filter's dynamic metadata
168    #[serde(rename = "metadataKey", default = "default_metadata_key")]
169    pub metadata_key: metadata::Key,
170}
171
172/// Default value for [`Config::metadata_key`]
173fn default_metadata_key() -> metadata::Key {
174    metadata::Key::from_static(CAPTURED_BYTES)
175}
176
177impl Default for Config {
178    fn default() -> Self {
179        Self {
180            metadata_key: default_metadata_key(),
181        }
182    }
183}
184
185impl From<Config> for proto::TokenRouter {
186    fn from(config: Config) -> Self {
187        Self {
188            metadata_key: Some(config.metadata_key.to_string()),
189        }
190    }
191}
192
193impl TryFrom<proto::TokenRouter> for Config {
194    type Error = ConvertProtoConfigError;
195
196    fn try_from(p: proto::TokenRouter) -> Result<Self, Self::Error> {
197        Ok(Self {
198            metadata_key: p
199                .metadata_key
200                .map_or_else(default_metadata_key, metadata::Key::new),
201        })
202    }
203}

216    #[test]
217    fn convert_proto_config() {
218        let test_cases = vec![
219            (
220                "should succeed when all valid values are provided",
221                proto::TokenRouter {
222                    metadata_key: Some("foobar".into()),
223                },
224                Some(Config {
225                    metadata_key: "foobar".into(),
226                }),
227            ),
228            (
229                "should use correct default values",
230                proto::TokenRouter { metadata_key: None },
231                Some(Config {
232                    metadata_key: default_metadata_key(),
233                }),
234            ),
235        ];
236        for (name, proto_config, expected) in test_cases {
237            let result = Config::try_from(proto_config);
238            assert_eq!(
239                result.is_err(),
240                expected.is_none(),
241                "{}: error expectation does not match",
242                name
243            );
244            if let Some(expected) = expected {
245                assert_eq!(expected, result.unwrap(), "{name}");
246            }
247        }
248    }

317    fn with_ctx(
318        dest: &mut Vec<crate::net::EndpointAddress>,
319        test: impl FnOnce(ReadContext<'_, crate::collections::PoolBuffer>),
320    ) {
321        let endpoint1 = Endpoint::with_metadata(
322            "127.0.0.1:80".parse().unwrap(),
323            Metadata {
324                tokens: vec!["123".into()].into_iter().collect(),
325            },
326        );
327        let endpoint2 = Endpoint::with_metadata(
328            "127.0.0.1:90".parse().unwrap(),
329            Metadata {
330                tokens: vec!["456".into()].into_iter().collect(),
331            },
332        );
333
334        let pool = std::sync::Arc::new(crate::collections::BufferPool::new(1, 5));
335
336        let endpoints = crate::net::cluster::ClusterMap::default();
337        endpoints.insert_default([endpoint1, endpoint2].into());
338        test(ReadContext::new(
339            &endpoints,
340            "127.0.0.1:100".parse().unwrap(),
341            pool.alloc_slice(b"hello"),
342            dest,
343        ));
344    }
345
346    fn assert_read<F, P>(filter: &F, mut ctx: ReadContext<'_, P>)
347    where
348        F: Filter + ?Sized,
349        P: PacketMut,
350    {
351        filter.read(&mut ctx).unwrap();
352        assert_eq!(b"hello", ctx.contents.as_slice());
353    }
