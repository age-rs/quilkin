17use crate::filters::prelude::*;
18use serde::{Deserialize, Serialize};
19
20use crate::generated::quilkin::filters::drop::v1alpha1 as proto;
21
22pub const NAME: &str = Drop::NAME;
23
24/// Always drops a packet, mostly useful in combination with other filters.
25pub struct Drop;
26
27impl Drop {
28    fn new() -> Self {
29        Self
30    }
31}
32
33impl Filter for Drop {
34    #[cfg_attr(feature = "instrument", tracing::instrument(skip_all))]
35    fn read<P: PacketMut>(&self, _: &mut ReadContext<'_, P>) -> Result<(), FilterError> {
36        Err(FilterError::Dropped)
37    }
38
39    #[cfg_attr(feature = "instrument", tracing::instrument(skip_all))]
40    fn write<P: PacketMut>(&self, _: &mut WriteContext<P>) -> Result<(), FilterError> {
41        Err(FilterError::Dropped)
42    }
43}
44
45impl StaticFilter for Drop {
46    const NAME: &'static str = "quilkin.filters.drop.v1alpha1.Drop";
47    type Configuration = Config;
48    type BinaryConfiguration = proto::Drop;
49
50    fn try_from_config(_: Option<Self::Configuration>) -> Result<Self, CreationError> {
51        Ok(Drop::new())
52    }
53}
54
55/// `pass` filter's configuration.
56#[derive(Serialize, Deserialize, Debug, schemars::JsonSchema)]
57pub struct Config;
58
59impl From<Config> for proto::Drop {
60    fn from(_config: Config) -> Self {
61        Self {}
62    }
63}
64
65impl TryFrom<proto::Drop> for Config {
66    type Error = ConvertProtoConfigError;
67
68    fn try_from(_: proto::Drop) -> Result<Self, Self::Error> {
69        Ok(Config)
70    }
71}
