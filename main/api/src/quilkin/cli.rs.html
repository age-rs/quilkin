49#[command(next_help_heading = "Locality Options")]
50pub struct LocalityCli {
51    /// The `region` to set in the cluster map for any provider
52    /// endpoints discovered.
53    #[clap(long = "locality.region", env = "QUILKIN_LOCALITY_REGION")]
54    pub region: Option<String>,
55    /// The `zone` in the `region` to set in the cluster map for any provider
56    /// endpoints discovered.
57    #[clap(
58        long = "locality.region.zone",
59        requires("region"),
60        env = "QUILKIN_LOCALITY_ZONE"
61    )]
62    pub zone: Option<String>,
63    /// The `sub_zone` in the `zone` in the `region` to set in the cluster map
64    /// for any provider endpoints discovered.
65    #[clap(
66        long = "locality.region.sub-zone",
67        requires("zone"),
68        env = "QUILKIN_LOCALITY_SUB_ZONE"
69    )]
70    pub sub_zone: Option<String>,
71    /// The airport ICAO code for the `region` of the instance. When provided
72    /// enables geomapping metrics of instances.
73    #[clap(
74        long = "locality.icao",
75        env = "QUILKIN_LOCALITY_ICAO",
76        default_value_t = crate::config::IcaoCode::default()
77    )]
78    pub icao_code: crate::config::IcaoCode,
79}

93/// Quilkin: a non-transparent UDP proxy specifically designed for use with
94/// large scale multiplayer dedicated game servers deployments, to
95/// ensure security, access control, telemetry data, metrics and more.
96#[derive(Debug, clap::Parser)]
97#[command(version)]
98#[non_exhaustive]
99pub struct Cli {
100    /// The path to the configuration file for the Quilkin instance.
101    #[clap(short, long, env = "QUILKIN_CONFIG", default_value = "quilkin.yaml")]
102    pub config: PathBuf,
103    /// Whether Quilkin will report any results to stdout/stderr.
104    #[clap(short, long, env)]
105    pub quiet: bool,
106    #[clap(subcommand)]
107    pub command: Option<Commands>,
108    #[clap(
109     long,
110     default_value_t = LogFormats::Auto,
111     value_parser = clap::builder::PossibleValuesParser::new(["auto", "json", "plain", "pretty"])
112     .map(|s| s.parse::<LogFormats>().unwrap()),
113     )]
114    pub log_format: LogFormats,
115    /// The file prefix used for log files
116    #[clap(
117        long = "sys.log.file-prefix",
118        env = "QUILKIN_SYS_LOG_FILE_PREFIX",
119        default_value = "quilkin.log"
120    )]
121    pub log_file_prefix: String,
122    /// An optional log file directory path that quilkin should log to
123    #[clap(long = "sys.log.dir", env = "QUILKIN_SYS_LOG_DIRECTORY")]
124    pub log_directory: Option<PathBuf>,
125    #[command(flatten)]
126    pub admin: AdminCli,
127    #[command(flatten)]
128    pub locality: LocalityCli,
129    #[command(flatten)]
130    pub providers: crate::Providers,
131    #[command(flatten)]
132    pub service: crate::service::Service,
133}

135/// The various log format options
136#[derive(Copy, Clone, PartialEq, Eq, Debug, EnumString, Display, Default)]
137pub enum LogFormats {
138    #[strum(serialize = "auto")]
139    #[default]
140    Auto,
141    #[strum(serialize = "json")]
142    Json,
143    #[strum(serialize = "plain")]
144    Plain,
145    #[strum(serialize = "pretty")]
146    Pretty,
147}
148
149impl LogFormats {
150    /// Creates the tracing subscriber that pulls from the env for filters
151    /// and outputs based on [Self].
152    fn init_tracing_subscriber(
153        self,
154        quiet: bool,
155        file_writer: Option<tracing_appender::non_blocking::NonBlocking>,
156    ) {
157        use tracing_subscriber::fmt::writer::{BoxMakeWriter, MakeWriterExt};
158
159        let env_filter = tracing_subscriber::EnvFilter::builder()
160            .with_default_directive(tracing_subscriber::filter::LevelFilter::INFO.into())
161            .from_env_lossy();
162
163        let mk_writer: BoxMakeWriter = match file_writer {
164            Some(file_writer) => {
165                if quiet {
166                    BoxMakeWriter::new(file_writer)
167                } else {
168                    BoxMakeWriter::new(std::io::stdout.and(file_writer))
169                }
170            }
171            None => {
172                if quiet {
173                    BoxMakeWriter::new(std::io::sink)
174                } else {
175                    BoxMakeWriter::new(std::io::stdout)
176                }
177            }
178        };
179
180        let subscriber = tracing_subscriber::fmt()
181            .with_file(true)
182            .with_thread_ids(true)
183            .with_env_filter(env_filter)
184            .with_writer(mk_writer);
185
186        match self {
187            LogFormats::Auto => {
188                use std::io::IsTerminal;
189                if !std::io::stdout().is_terminal() {
190                    subscriber.with_ansi(false).json().init();
191                } else {
192                    subscriber.init();
193                }
194            }
195            LogFormats::Json => subscriber.with_ansi(false).json().init(),
196            LogFormats::Plain => subscriber.init(),
197            LogFormats::Pretty => subscriber.pretty().init(),
198        }
199    }
200}

210impl Cli {
211    /// Drives the main quilkin application lifecycle using the command line
212    /// arguments.
213    #[tracing::instrument(skip_all)]
214    pub async fn drive(self) -> crate::Result<()> {
215        // Configure rolling log file appender if directory has been specified.
216        // _log_file_guard should be kept in scope and will trigger the final flush to file when dropped
217        let (file_writer, _log_file_guard) = match &self.log_directory {
218            Some(log_directory) => {
219                let file_appender = tracing_appender::rolling::Builder::new()
220                    .rotation(tracing_appender::rolling::Rotation::HOURLY)
221                    .filename_prefix(&self.log_file_prefix)
222                    .max_log_files(5)
223                    .build(log_directory)
224                    .expect("failed to build rolling file appender");
225                let (file_writer, guard) = tracing_appender::non_blocking(file_appender);
226                (Some(file_writer), Some(guard))
227            }
228            None => (None, None),
229        };
230
231        self.log_format
232            .init_tracing_subscriber(self.quiet, file_writer);
233
234        tracing::info!(
235            version = crate_version!(),
236            commit = crate::net::endpoint::metadata::build::GIT_COMMIT_HASH,
237            "Starting Quilkin"
238        );
239
240        // Non-long running commands (e.g. ones with no administration server)
241        // are executed here.
242        match self.command {
243            Some(Commands::Qcmp(Qcmp::Ping(ping))) => return ping.run().await,
244            Some(Commands::GenerateConfigSchema(generator)) => {
245                return generator.generate_config_schema();
246            }
247            _ => {}
248        }
249
250        if !self.service.any_service_enabled()
251            && self.command.is_none()
252            && !self.providers.any_provider_enabled()
253        {
254            eyre::bail!("no service, provider, or command specified, shutting down");
255        }
256
257        tracing::debug!(cli = ?self, "config parameters");
258
259        let locality = self.locality.locality();
260        let shutdown_handler = crate::signal::spawn_handler();
261        let drive_token = crate::signal::cancellation_token(shutdown_handler.shutdown_rx());
262
263        let config = crate::Config::new_rc(
264            self.service.id.clone(),
265            self.locality.icao_code,
266            &self.providers,
267            &self.service,
268            drive_token.child_token(),
269        );
270        config.read_config(&self.config, locality.clone())?;
271
272        crate::metrics::with_mut_registry(|mut registry| {
273            crate::metrics::register_metrics(&mut registry, config.id());
274        });
275
276        let ready = Arc::<std::sync::atomic::AtomicBool>::default();
277        if self.admin.enabled {
278            crate::components::admin::serve(
279                config.clone(),
280                ready.clone(),
281                shutdown_handler.shutdown_tx(),
282                shutdown_handler.shutdown_rx(),
283                self.admin.address,
284            );
285        }
286
287        crate::alloc::spawn_heap_stats_updates(
288            std::time::Duration::from_secs(10),
289            shutdown_handler.shutdown_rx(),
290        );
291
292        // Just call this early so there isn't a potential race when spawning xDS
293        quilkin_xds::metrics::set_registry(crate::metrics::registry());
294
295        let mut provider_tasks = self.providers.spawn_providers(
296            &config,
297            ready.clone(),
298            locality.clone(),
299            None,
300            shutdown_handler.shutdown_rx(),
301        );
302
303        let shutdown_tx = shutdown_handler.shutdown_tx();
304        let mut service_task = self.service.spawn_services(&config, shutdown_handler)?;
305
306        if provider_tasks.is_empty() {
307            ready.store(true, std::sync::atomic::Ordering::SeqCst);
308        }
309
310        loop {
311            tokio::select! {
312                Some(result) = provider_tasks.join_next() => {
313                    match result {
314                        Ok(_) => {
315                            // TODO should improve the provider tasks shutdown so we can log
316                            // exactly which provider has stopped
317                            tracing::info!("provider task stopped");
318                        },
319                        Err(error) => {
320                            tracing::error!(task_result=?error, "provider task completed unexpectedly, shutting down.");
321                            // Trigger shutdown so we can drain the active sessions in the service_task
322                            if let Err(error) = shutdown_tx.send(()) {
323                                tracing::error!(error=?error, "failed to trigger shutdown");
324                                return Err(error.into());
325                            }
326                        },
327                    }
328                },
329                result = &mut service_task => {
330                    return match result {
331                        Ok((_, result)) => {
332                            result
333                        }
334                        Err(join_error) => {
335                            Err(eyre::format_err!("failed to join services task: {join_error}"))
336                        }
337                    };
338                },
339            }
340        }
341    }
342}

344#[derive(Debug, Copy, Clone)]
345pub struct Duration(std::time::Duration);
346
347impl std::str::FromStr for Duration {
348    type Err = clap::Error;
349
350    fn from_str(src: &str) -> Result<Self, Self::Err> {
351        let suffix_pos = src.find(char::is_alphabetic).unwrap_or(src.len());
352
353        let num: u64 = src[..suffix_pos]
354            .parse()
355            .map_err(|err| clap::Error::raw(clap::error::ErrorKind::ValueValidation, err))?;
356        let suffix = if suffix_pos == src.len() {
357            "s"
358        } else {
359            &src[suffix_pos..]
360        };
361        use std::time::Duration as D;
362        let duration = match suffix {
363            "ms" | "MS" => D::from_millis(num),
364            "s" | "S" => D::from_secs(num),
365            "m" | "M" => D::from_secs(num * 60),
366            "h" | "H" => D::from_secs(num * 60 * 60),
367            "d" | "D" => D::from_secs(num * 60 * 60 * 24),
368            s => {
369                return Err(clap::Error::raw(
370                    clap::error::ErrorKind::ValueValidation,
371                    format!("unknown duration suffix '{s}'"),
372                ));
373            }
374        };
375
376        Ok(Self(duration))
377    }
378}
379
380impl std::ops::Deref for Duration {
381    type Target = std::time::Duration;
382
383    fn deref(&self) -> &Self::Target {
384        &self.0
385    }
