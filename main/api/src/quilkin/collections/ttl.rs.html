36/// A wrapper around the value of an entry in the map.
37/// It contains the value's ttl.
38pub struct Value<V> {
39    pub value: V,
40    expires_at: Arc<AtomicU64>,
41    clock: Clock,
42}
43
44impl<V> Value<V> {
45    fn new(value: V, ttl: Duration, clock: Clock) -> Value<V> {
46        let value = Value {
47            value,
48            expires_at: Arc::new(AtomicU64::new(0)),
49            clock,
50        };
51        value.update_expiration(ttl);
52        value
53    }
54
55    /// Get the expiration time for this value. The returned value is the
56    /// number of seconds relative to some reference point (e.g `UNIX_EPOCH`), based
57    /// on the clock being used.
58    #[inline]
59    fn expiration_secs(&self) -> u64 {
60        self.expires_at.load(Ordering::Relaxed)
61    }
62
63    /// Update the value's expiration time to (now + TTL).
64    #[inline]
65    fn update_expiration(&self, ttl: Duration) {
66        match self.clock.compute_expiration_secs(ttl) {
67            Ok(new_expiration_time) => {
68                self.expires_at
69                    .store(new_expiration_time, Ordering::Relaxed);
70            }
71            Err(err) => {
72                warn!("failed to increment key expiration: {}", err);
73            }
74        }
75    }
76}
77
78impl<V: std::fmt::Debug> std::fmt::Debug for Value<V> {
79    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
80        f.debug_struct("Value")
81            .field("value", &self.value)
82            .field("expires_at", &self.expires_at)
83            .finish()
84    }
85}

87impl<V> std::ops::Deref for Value<V> {
88    type Target = V;
89
90    fn deref(&self) -> &Self::Target {
91        &self.value
92    }

123/// A key value hash map where entries are associated with a TTL.
124///
125/// When the TTL for an entry elapses, the entry is removed from the map.
126///
127/// The TTL is reset each time the entry is (re)inserted or read via [`TtlMap::get`],
128/// [`TtlMap::get_mut`] functions, or via the [`TtlMap::entry`] interface.
129///
130/// During tests, the internal clock implementation is driven by [`tokio::time`] so
131/// functions like [`tokio::time::pause`] and [`tokio::time::advance`] can be used.
132pub struct TtlMap<K, V>(Arc<Map<K, V>>);
133
134impl<K, V> TtlMap<K, V>
135where
136    K: Hash + Eq + Send + Sync + 'static,
137    V: Send + Sync + 'static,
138{
139    pub fn new(ttl: Duration, poll_interval: Duration) -> Self {
140        Self::initialize(DashMap::new(), ttl, poll_interval)
141    }
142
143    #[allow(dead_code)]
144    pub fn with_capacity(ttl: Duration, poll_interval: Duration, capacity: usize) -> Self {
145        Self::initialize(DashMap::with_capacity(capacity), ttl, poll_interval)
146    }
147
148    fn initialize(inner: DashMap<K, Value<V>>, ttl: Duration, poll_interval: Duration) -> Self {
149        let (shutdown_tx, shutdown_rx) = channel();
150        let map = TtlMap(Arc::new(Map {
151            inner,
152            shutdown_tx: Some(shutdown_tx),
153            ttl,
154            clock: Clock::new(),
155        }));
156        spawn_cleanup_task(
157            map.0.clone(),
158            poll_interval,
159            map.0.clock.clone(),
160            shutdown_rx,
161        );
162        map
163    }
164
165    /// Returns the current time as the number of seconds relative to some initial
166    /// reference point (e.g `UNIX_EPOCH`), based on the clock implementation being used.
167    /// In tests, this will be driven by [`tokio::time`]
168    #[inline]
169    pub(crate) fn now_relative_secs(&self) -> u64 {
170        self.0.clock.now_relative_secs().unwrap_or_default()
171    }
172}

174#[allow(dead_code)]
175impl<K, V> TtlMap<K, V>
176where
177    K: Hash + Eq + Send + Sync + 'static,
178    V: Send + Sync,
179{
180    /// Returns a reference to value corresponding to key.
181    #[inline]
182    pub fn get(&self, key: &K) -> Option<Ref<'_, K, Value<V>>> {
183        let value = self.0.inner.get(key);
184        if let Some(value) = &value {
185            value.update_expiration(self.0.ttl);
186        }
187
188        value
189    }
190
191    /// Returns a reference to value corresponding to key.
192    #[inline]
193    pub fn try_get(&self, key: &K) -> TryResult<Ref<'_, K, Value<V>>> {
194        let value = self.0.inner.try_get(key);
195        if let TryResult::Present(value) = &value {
196            value.update_expiration(self.0.ttl);
197        }
198
199        value
200    }
201
202    /// Returns a mutable reference to value corresponding to key.
203    /// The value will be reset to expire at the configured TTL after the time of retrieval.
204    #[inline]
205    pub fn get_mut(&self, key: &K) -> Option<RefMut<'_, K, Value<V>>> {
206        let value = self.0.inner.get_mut(key);
207        if let Some(ref value) = value {
208            value.update_expiration(self.0.ttl);
209        }
210
211        value
212    }
213
214    /// Returns the number of entries currently in the map.
215    #[inline]
216    pub fn len(&self) -> usize {
217        self.0.inner.len()
218    }
219
220    /// Returns whether the map currently contains no entries.
221    #[inline]
222    pub fn is_empty(&self) -> bool {
223        self.len() == 0
224    }
225
226    /// Returns whether the map currently contains any entries.
227    #[inline]
228    pub fn is_not_empty(&self) -> bool {
229        !self.is_empty()
230    }
231
232    /// Returns true if the map contains a value for the specified key.
233    #[inline]
234    pub fn contains_key(&self, key: &K) -> bool {
235        self.0.inner.contains_key(key)
236    }
237
238    /// Inserts a key-value pair into the map.
239    /// The value will be set to expire at the configured TTL after the time of insertion.
240    /// If a previous value existed for this key, that value is returned.
241    #[inline]
242    pub fn insert(&self, key: K, value: V) -> Option<V> {
243        self.0
244            .inner
245            .insert(key, Value::new(value, self.0.ttl, self.0.clock.clone()))
246            .map(|value| value.value)
247    }
248
249    /// Removes a key-value pair from the map.
250    #[inline]
251    pub fn remove(&self, key: K) -> bool {
252        self.0.inner.remove(&key).is_some()
253    }
254
255    /// Removes all entries from the map
256    #[inline]
257    pub fn clear(&self) {
258        self.0.inner.clear();
259    }
260
261    /// Returns an entry for in-place updates of the specified key-value pair.
262    /// Note: This acquires a write lock on the map's shard that corresponds
263    /// to the entry.
264    #[inline]
265    pub fn entry(&self, key: K) -> Entry<'_, K, Value<V>> {
266        let ttl = self.0.ttl;
267        match self.0.inner.entry(key) {
268            inner @ DashMapEntry::Occupied(_) => Entry::Occupied(OccupiedEntry {
269                inner,
270                ttl,
271                clock: self.0.clock.clone(),
272            }),
273            inner @ DashMapEntry::Vacant(_) => Entry::Vacant(VacantEntry {
274                inner,
275                ttl,
276                clock: self.0.clock.clone(),
277            }),
278        }
279    }
280}

322/// A view into an entry in the map.
323/// It may either be [`VacantEntry`] or [`OccupiedEntry`]
324pub enum Entry<'a, K, V> {
325    Occupied(OccupiedEntry<'a, K, V>),
326    Vacant(VacantEntry<'a, K, V>),
327}
328
329impl<K, V> OccupiedEntry<'_, K, Value<V>>
330where
331    K: Eq + Hash,
332{
333    /// Returns a reference to the entry's value.
334    /// The value will be reset to expire at the configured TTL after the time of retrieval.
335    pub fn get(&self) -> &Value<V> {
336        match &self.inner {
337            DashMapEntry::Occupied(entry) => {
338                let value = entry.get();
339                value.update_expiration(self.ttl);
340                value
341            }
342            DashMapEntry::Vacant(_) => unreachable!("BUG: entry type should be occupied"),
343        }
344    }
345
346    #[allow(dead_code)]
347    /// Returns a mutable reference to the entry's value.
348    /// The value will be reset to expire at the configured TTL after the time of retrieval.
349    pub fn get_mut(&mut self) -> &mut Value<V> {
350        match &mut self.inner {
351            DashMapEntry::Occupied(entry) => {
352                let value = entry.get_mut();
353                value.update_expiration(self.ttl);
354                value
355            }
356            DashMapEntry::Vacant(_) => unreachable!("BUG: entry type should be occupied"),
357        }
358    }
359
360    #[allow(dead_code)]
361    /// Replace the entry's value with a new value, returning the old value.
362    /// The value will be set to expire at the configured TTL after the time of insertion.
363    pub fn insert(&mut self, value: V) -> Value<V> {
364        match &mut self.inner {
365            DashMapEntry::Occupied(entry) => {
366                entry.insert(Value::new(value, self.ttl, self.clock.clone()))
367            }
368            DashMapEntry::Vacant(_) => unreachable!("BUG: entry type should be occupied"),
369        }
370    }
371}

373impl<'a, K, V> VacantEntry<'a, K, Value<V>>
374where
375    K: Eq + Hash,
376{
377    /// Set an entry's value.
378    /// The value will be set to expire at the configured TTL after the time of insertion.
379    pub fn insert(self, value: V) -> RefMut<'a, K, Value<V>> {
380        match self.inner {
381            DashMapEntry::Vacant(entry) => {
382                entry.insert(Value::new(value, self.ttl, self.clock.clone()))
383            }
384            DashMapEntry::Occupied(_) => unreachable!("BUG: entry type should be vacant"),
385        }
386    }

389fn spawn_cleanup_task<K, V>(
390    map: Arc<Map<K, V>>,
391    poll_interval: Duration,
392    clock: Clock,
393    mut shutdown_rx: Receiver<()>,
394) where
395    K: Send + Sync + Hash + Eq + 'static,
396    V: Send + Sync + 'static,
397{
398    let mut interval = tokio::time::interval(poll_interval);
399
400    tokio::spawn(async move {
401        loop {
402            tokio::select! {
403                _ = interval.tick() => {
404                    prune_entries( &map, &clock).await;
405                }
406                _ = &mut shutdown_rx => {
407                    return;
408                }
409            }
410        }
411    });
412}
413
414async fn prune_entries<K, V>(map: &Arc<Map<K, V>>, clock: &Clock)
415where
416    K: Hash + Eq + Send + Sync + 'static,
417    V: Send + Sync + 'static,
418{
419    let now_secs = if let Ok(now_secs) = clock.now_relative_secs() {
420        now_secs
421    } else {
422        warn!("Failed to get current time when pruning sessions");
423        return;
424    };
425
426    // Take a read lock first and check if there is at least 1 item to remove.
427    let has_expired_keys = map
428        .inner
429        .iter()
430        .filter(|entry| entry.value().expiration_secs() <= now_secs)
431        .take(1)
432        .next()
433        .is_some();
434
435    // If we have work to do then, take a write lock.
436    if has_expired_keys {
437        // Go over the whole map in case anything expired
438        // since acquiring the write lock.
439        map.inner
440            .retain(|_, value| value.expiration_secs() > now_secs);
441    }
442}

444/// A wrapper over functions to generate relative timestamps and ttl.
445/// During test it is driven via [`tokio::time`], otherwise it uses system time.
446#[derive(Clone)]
447struct Clock {
448    #[cfg(test)]
449    base: Instant,
450}
451
452impl Clock {
453    fn new() -> Clock {
454        #[cfg(not(test))]
455        return Clock {};
456
457        #[cfg(test)]
458        return Clock {
459            base: Instant::now(),
460        };
461    }
462
463    /// Returns the current time in seconds, relative to some base time instant.
464    /// For non test cases, relative to `UNIX_EPOCH`, while during test, a random
465    /// point in the past is used.
466    #[cfg_attr(not(test), allow(clippy::unused_self))]
467    fn now_relative_secs(&self) -> Result<u64, String> {
468        #[cfg(not(test))]
469        return SystemTime::now()
470            .duration_since(UNIX_EPOCH)
471            .map_err(|_err| {
472                String::from("duration_since was called with time later than the current time")
473            })
474            .map(|t| t.as_secs());
475
476        #[cfg(test)]
477        return Ok((Instant::now()).duration_since(self.base).as_secs());
478    }
479
480    /// Returns the expiration time from now in seconds for the given ttl.
481    #[cfg_attr(not(test), allow(clippy::unused_self))]
482    fn compute_expiration_secs(&self, ttl: Duration) -> Result<u64, String> {
483        #[cfg(not(test))]
484        return (SystemTime::now() + ttl)
485            .duration_since(UNIX_EPOCH)
486            .map_err(|_err| {
487                String::from("duration_since was called with time later than the current time")
488            })
489            .map(|t| t.as_secs());
490
491        #[cfg(test)]
492        return Ok((Instant::now() + ttl).duration_since(self.base).as_secs());
493    }
494}

777    #[tokio::test]
778    async fn cleanup_expired_entries() {
779        // Test that we delete expired entries from the ttl map.
780        time::pause();
781
782        let (one, two) = address_pair();
783
784        let map =
785            TtlMap::<EndpointAddress, usize>::new(Duration::from_secs(5), Duration::from_secs(1));
786        map.insert(one.clone(), 1);
787        map.insert(two.clone(), 2);
788
789        assert!(map.contains_key(&one));
790        assert!(map.contains_key(&two));
791
792        time::advance(Duration::from_secs(4)).await;
793
794        // Read one key so that it does not expire at the original ttl.
795        map.get(&two).unwrap();
796
797        // Check that only the un-read key is deleted.
798        time::advance(Duration::from_secs(4)).await;
799        assert!(!map.contains_key(&one));
800        assert!(map.contains_key(&two));
801        assert_eq!(map.len(), 1);
802
803        // Check that the second key is eventually deleted.
804        time::advance(Duration::from_secs(3)).await;
805        assert!(!map.contains_key(&one));
806        assert!(!map.contains_key(&two));
807        assert_eq!(map.len(), 0);
808    }
