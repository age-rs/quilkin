32/// Representation of an immutable set of bytes pulled from the network, this trait
33/// provides an abstraction over however the packet was received (epoll, io-uring, xdp)
34///
35/// Use [`PacketMut`] if you need a mutable representation.
36pub trait Packet: Sized {
37    /// Returns the underlying slice of bytes representing the packet.
38    fn as_slice(&self) -> &[u8];
39
40    /// Returns the size of the packet.
41    fn len(&self) -> usize;
42
43    /// Returns whether the given packet is empty.
44    fn is_empty(&self) -> bool {
45        self.len() == 0
46    }
47}
48
49/// Representation of an mutable set of bytes pulled from the network, this trait
50/// provides an abstraction over however the packet was received (epoll, io-uring, xdp)
51pub trait PacketMut: Sized + Packet {
52    type FrozenPacket: Packet;
53    fn remove_head(&mut self, length: usize);
54    fn remove_tail(&mut self, length: usize);
55    fn extend_head(&mut self, bytes: &[u8]);
56    fn extend_tail(&mut self, bytes: &[u8]);
57    /// Returns an immutable version of the packet, this allows certain types
58    /// return a type that can be more cheaply cloned and shared.
59    fn freeze(self) -> Self::FrozenPacket;
60}

62/// Packet received from local port
63pub(crate) struct DownstreamPacket<'stack, P> {
64    pub(crate) contents: P,
65    pub(crate) source: SocketAddr,
66    pub(crate) filters: &'stack crate::filters::FilterChain,
67}
68
69impl<P: PacketMut> DownstreamPacket<'_, P> {
70    #[inline]
71    pub(crate) fn process<S: SessionManager<Packet = P::FrozenPacket>>(
72        self,
73        worker_id: usize,
74        config: &Arc<Config>,
75        sessions: &S,
76        destinations: &mut Vec<crate::net::EndpointAddress>,
77    ) {
78        tracing::trace!(
79            id = worker_id,
80            size = self.contents.len(),
81            source = %self.source,
82            "received packet from downstream"
83        );
84
85        let timer = metrics::processing_time(metrics::READ).start_timer();
86        if let Err(error) = self.process_inner(config, sessions, destinations) {
87            let discriminant = error.discriminant();
88
89            error.inc_system_errors_total(metrics::READ, &metrics::EMPTY);
90            metrics::packets_dropped_total(metrics::READ, discriminant, &metrics::EMPTY).inc();
91        }
92
93        timer.stop_and_record();
94    }
95
96    /// Processes a packet by running it through the filter chain.
97    #[inline]
98    fn process_inner<S: SessionManager<Packet = P::FrozenPacket>>(
99        self,
100        config: &Arc<Config>,
101        sessions: &S,
102        destinations: &mut Vec<crate::net::EndpointAddress>,
103    ) -> Result<(), PipelineError> {
104        let Some(clusters) = config
105            .dyn_cfg
106            .clusters()
107            .filter(|c| c.read().has_endpoints())
108        else {
109            tracing::trace!("no upstream endpoints");
110            return Err(PipelineError::NoUpstreamEndpoints);
111        };
112
113        #[cfg(not(debug_assertions))]
114        {
115            match self.source.ip() {
116                std::net::IpAddr::V4(ipv4) => {
117                    if ipv4.is_loopback() || ipv4.is_multicast() || ipv4.is_broadcast() {
118                        return Err(PipelineError::DisallowedSourceIP(self.source.ip()));
119                    }
120                }
121                std::net::IpAddr::V6(ipv6) => {
122                    if ipv6.is_loopback() || ipv6.is_multicast() {
123                        return Err(PipelineError::DisallowedSourceIP(self.source.ip()));
124                    }
125                }
126            }
127        }
128
129        let cm = clusters.clone_value();
130        let mut context = ReadContext::new(&cm, self.source.into(), self.contents, destinations);
131        self.filters
132            .read(&mut context)
133            .map_err(PipelineError::Filter)?;
134
135        let ReadContext { contents, .. } = context;
136
137        // Similar to bytes::BytesMut::freeze, we turn the mutable pool buffer
138        // into an immutable one with its own internal arc so it can be cloned
139        // cheaply and returned to the pool once all references are dropped
140        let contents = contents.freeze();
141
142        for epa in destinations.drain(0..) {
143            let session_key = SessionKey {
144                source: self.source,
145                dest: epa.to_socket_addr()?,
146            };
147
148            sessions.send(session_key, &contents)?;
149        }
150
151        Ok(())
152    }
153}

155/// Spawns a background task that sits in a loop, receiving packets from the passed in socket.
156/// Each received packet is placed on a queue to be processed by a worker task.
157/// This function also spawns the set of worker tasks responsible for consuming packets
158/// off the aforementioned queue and processing them through the filter chain and session
159/// pipeline.
160pub fn spawn_receivers(
161    config: Arc<Config>,
162    socket: socket2::Socket,
163    worker_sends: Vec<crate::net::PacketQueue>,
164    sessions: &Arc<SessionPool>,
165    buffer_pool: Arc<crate::collections::BufferPool>,
166) -> crate::Result<()> {
167    let port = crate::net::socket_port(&socket);
168
169    let Some(pfc) = config.dyn_cfg.cached_filter_chain() else {
170        eyre::bail!("the ProxyFilterChain state was not configured");
171    };
172
173    for (worker_id, ws) in worker_sends.into_iter().enumerate() {
174        let worker = crate::net::io::Listener {
175            worker_id,
176            port,
177            config: config.clone(),
178            sessions: sessions.clone(),
179            buffer_pool: buffer_pool.clone(),
180        };
181
182        worker.spawn_io_loop(ws, pfc.clone())?;
183    }
184
185    Ok(())
186}

188#[cfg(test)]
189mod tests {
190    #![cfg(not(debug_assertions))]
191
192    use quilkin_xds::locality::Locality;
193
194    use crate::collections::BufferPool;
195    use crate::net::Endpoint;
196    use crate::test::alloc_buffer;
197
198    use super::*;
199    use std::net::{IpAddr, Ipv4Addr, Ipv6Addr};
200    use std::net::{SocketAddrV4, SocketAddrV6};
201
202    // Ensure we disallow certain source IP addresses to protect against UDP amplification attacks
203    #[tokio::test]
204    async fn disallowed_ips() {
205        let nl1 = Locality::with_region("nl-1");
206        let endpoint = Endpoint::new((Ipv4Addr::LOCALHOST, 7777).into());
207
208        let config = Arc::new(Config::default_agent().cluster(
209            None,
210            Some(nl1.clone()),
211            [endpoint.clone()].into(),
212        ));
213        let buffer_pool = Arc::new(BufferPool::new(1, 10));
214        let session_manager = SessionPool::new(config.clone(), vec![], buffer_pool.clone());
215
216        let packet_data: [u8; 4] = [1, 2, 3, 4];
217        for ip in [
218            IpAddr::V4(Ipv4Addr::LOCALHOST),
219            IpAddr::V4(Ipv4Addr::BROADCAST),
220            // multicast = 224.0.0.0/4
221            IpAddr::V4(Ipv4Addr::new(224, 0, 0, 0)),
222            IpAddr::V4(Ipv4Addr::new(239, 255, 255, 255)),
223            IpAddr::V6(Ipv6Addr::LOCALHOST),
224            // multicast = any address starting with 0xff
225            IpAddr::V6(Ipv6Addr::new(0xff00, 0, 0, 0, 0, 0, 0, 0)),
226        ] {
227            let packet = DownstreamPacket {
228                contents: alloc_buffer(packet_data),
229                source: match ip {
230                    IpAddr::V4(ipv4) => SocketAddr::V4(SocketAddrV4::new(ipv4, 0)),
231                    IpAddr::V6(ipv6) => SocketAddr::V6(SocketAddrV6::new(ipv6, 0, 0, 0)),
232                },
233            };
234
235            let mut endpoints = vec![endpoint.address.clone()];
236            let res = packet.process_inner(&config, &session_manager, &mut endpoints);
237
238            assert_eq!(res, Err(PipelineError::DisallowedSourceIP(ip)));
239        }
240    }
