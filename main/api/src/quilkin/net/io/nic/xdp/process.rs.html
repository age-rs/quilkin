31/// Wrapper around the actual packet buffer and the UDP metadata it parsed to
32/// so that we can satisify the filter traits
33struct PacketWrapper {
34    buffer: Packet,
35    headers: UdpHeaders,
36}
37
38impl filters::Packet for PacketWrapper {
39    #[inline]
40    fn as_slice(&self) -> &[u8] {
41        &self.buffer[self.headers.data.start..self.headers.data.end]
42    }
43
44    #[inline]
45    fn len(&self) -> usize {
46        self.headers.data_length()
47    }
48}
49
50impl filters::PacketMut for PacketWrapper {
51    type FrozenPacket = PacketWrapper;
52
53    #[inline]
54    fn extend_head(&mut self, bytes: &[u8]) {
55        self.buffer
56            .insert(self.headers.data.start, bytes)
57            .expect("failed to extend head");
58        self.headers.data.end += bytes.len();
59    }
60
61    #[inline]
62    fn extend_tail(&mut self, bytes: &[u8]) {
63        self.buffer.append(bytes).expect("failed to extend head");
64        self.headers.data.end += bytes.len();
65    }
66
67    #[inline]
68    fn remove_head(&mut self, length: usize) {
69        let mut data = [0u8; 2048];
70
71        if length > self.headers.data_length() || length == 0 {
72            return;
73        }
74
75        let Some(slice) = self
76            .buffer
77            .get(self.headers.data.start + length..self.headers.data.end)
78        else {
79            if self.buffer.adjust_tail(-(length as i32)).is_ok() {
80                self.headers.data.end -= length;
81            }
82            return;
83        };
84        let remainder = slice.len();
85        data[..remainder].copy_from_slice(slice);
86
87        let Some(slice) = self
88            .buffer
89            .get_mut(self.headers.data.start..self.headers.data.start + remainder)
90        else {
91            return;
92        };
93        slice.copy_from_slice(&data[..remainder]);
94
95        if self.buffer.adjust_tail(-(length as i32)).is_ok() {
96            self.headers.data.end -= length;
97        }
98    }
99
100    #[inline]
101    fn remove_tail(&mut self, length: usize) {
102        if self.buffer.adjust_tail(-(length as i32)).is_ok() {
103            self.headers.data.end -= length;
104        }
105    }
106
107    // Only used in the io-uring implementation
108    fn freeze(self) -> Self::FrozenPacket {
109        unreachable!();
110    }
111}

134impl State {
135    /// Maps a remote server (upstream) endpoint back to the client endpoint
136    /// that initiated the session
137    #[inline]
138    fn lookup_client(
139        &self,
140        server_addr: SocketAddr,
141        port: NetworkU16,
142    ) -> Option<(SocketAddr, AsnInfo<'_>)> {
143        let addr = self.sessions.lookup_client(server_addr, port)?;
144        let entry = self
145            .addr_to_asn
146            .get(&addr.ip())
147            .and_then(|ipe| {
148                ipe.as_ref().map(|(ipe, asn)| AsnInfo {
149                    prefix: &ipe.prefix,
150                    asn: asn.as_str(),
151                })
152            })
153            .unwrap_or(metrics::EMPTY);
154
155        Some((addr, entry))
156    }
157
158    /// Retrieves or creates a session, ie a mapping of a server endpoint + port
159    /// to a client endpoint
160    #[inline]
161    fn session(
162        &mut self,
163        client_addr: SocketAddr,
164        server_addr: SocketAddr,
165    ) -> (NetworkU16, AsnInfo<'_>, IpAddresses) {
166        let ips = self.ips(server_addr.ip());
167        let asn = self.addr_to_asn.entry(client_addr.ip()).or_insert_with(|| {
168            let ipe = maxmind_db::MaxmindDb::lookup(client_addr.ip());
169            ipe.map(|ipe| {
170                let asn = maxmind_db::Asn::new(ipe.id);
171                (ipe, asn)
172            })
173        });
174
175        let port =
176            self.sessions
177                .get_or_create(client_addr, server_addr, asn.as_ref().map(|(ipe, _)| ipe));
178
179        (
180            port,
181            asn.as_ref().map_or(metrics::EMPTY, |(ipe, asn)| AsnInfo {
182                prefix: &ipe.prefix,
183                asn: asn.as_str(),
184            }),
185            ips,
186        )
187    }

189    #[inline]
190    fn ips(&self, destination: IpAddr) -> IpAddresses {
191        match destination {
192            IpAddr::V4(destination) => IpAddresses::V4 {
193                source: self.local_ipv4,
194                destination,
195            },
196            IpAddr::V6(destination) => IpAddresses::V6 {
197                source: self.local_ipv6,
198                destination,
199            },
200        }
201    }

204/// Linux by default only allocates ephemeral ports between 32768..=60999
205/// (see `/proc/sys/net/ipv4/ip_local_port_range`), so we take advantage and only
206/// allocate ports above that range. Note that we check that this range hasn't
207/// been modified during XDP initialization, if that changes the port mapping
208/// code could cause issues
209const EPHEMERAL_RANGE_END: u16 = 61000;
210/// With 18 bytes per address, this lets each bucket fit in < 2k
211const BUCKET_SIZE: usize = 112;
212
213#[repr(C)]
214struct Item {
215    octets: [u8; 16],
216    port: u16,
217}
218
219impl Item {
220    #[inline]
221    fn set(&mut self, addr: SocketAddr) {
222        match addr {
223            SocketAddr::V4(v4) => {
224                // We'll never be sending to multicast addresses, so use that
225                // fact to encode that this is an ipv4 address
226                self.octets[0] = 0xff;
227                self.octets[12..].copy_from_slice(&v4.ip().octets());
228            }
229            SocketAddr::V6(v6) => {
230                self.octets = v6.ip().octets();
231            }
232        }
233
234        self.port = addr.port();
235    }
236
237    #[inline]
238    fn get(&self) -> SocketAddr {
239        if self.octets[0] == 0xff {
240            (
241                std::net::Ipv4Addr::new(
242                    self.octets[12],
243                    self.octets[13],
244                    self.octets[14],
245                    self.octets[15],
246                ),
247                self.port,
248            )
249                .into()
250        } else {
251            (std::net::Ipv6Addr::from(self.octets), self.port).into()
252        }
253    }
254}
255
256struct PortMap {
257    buckets: Vec<[Item; BUCKET_SIZE]>,
258}
259
260impl PortMap {
261    #[inline]
262    fn new() -> Self {
263        Self {
264            // SAFETY: Item is POD
265            buckets: vec![unsafe { std::mem::zeroed() }],
266        }
267    }
268
269    #[inline]
270    fn get(&self, port: NetworkU16) -> Option<SocketAddr> {
271        let i = (port.host() - EPHEMERAL_RANGE_END) as usize;
272        let bucket = i / BUCKET_SIZE;
273
274        let bucket = self.buckets.get(bucket)?;
275
276        // SAFETY: We know the index is valid
277        unsafe {
278            let item = bucket.get_unchecked(i % BUCKET_SIZE);
279
280            // A zero port means this item was never initialized
281            if item.port == 0 {
282                return None;
283            }
284
285            Some(item.get())
286        }
287    }
288
289    #[inline]
290    fn insert(&mut self, client_addr: SocketAddr, port: u16) {
291        let i = (port - EPHEMERAL_RANGE_END) as usize;
292        let bucket = i / BUCKET_SIZE;
293        if self.buckets.len() == bucket {
294            // SAFETY: POD
295            self.buckets.push(unsafe { std::mem::zeroed() });
296        }
297
298        // SAFETY: We've guaranteed we have a bucket at the index, and the
299        // bucket has a fixed size of initialized bytes ready
300        unsafe {
301            self.buckets
302                .get_unchecked_mut(bucket)
303                .get_unchecked_mut(i % BUCKET_SIZE)
304                .set(client_addr);
305        }
306    }
307}

312    /// The port used to identify this unique session to the IP owning this map
313    port: NetworkU16,

316struct PortMapper {
317    /// Maps a client endpoint to the port used as the source port for sending
318    /// to the server endpoint `Self` is associated with
319    client_to_port: Arc<parking_lot::Mutex<std::collections::HashMap<SocketAddr, ClientInfo>>>,
320    port_to_client: Arc<parking_lot::RwLock<PortMap>>,
321    port: AtomicU16,
322}
323
324impl PortMapper {
325    #[inline]
326    fn new() -> Self {
327        Self {
328            client_to_port: Arc::new(Default::default()),
329            port_to_client: Arc::new(parking_lot::RwLock::new(PortMap::new())),
330            port: AtomicU16::new(EPHEMERAL_RANGE_END),
331        }
332    }
333
334    #[inline]
335    fn get_or_alloc(
336        &self,
337        client_addr: SocketAddr,
338        asn: Option<&IpNetEntry>,
339    ) -> Option<NetworkU16> {
340        match self.client_to_port.lock().entry(client_addr) {
341            Entry::Occupied(entry) => Some(entry.get().port),
342            Entry::Vacant(entry) => {
343                let port = self.port.fetch_add(1, Ordering::Relaxed);
344
345                if port < EPHEMERAL_RANGE_END {
346                    // This means we've overflowed
347                    return None;
348                }
349
350                session_metrics::total_sessions().inc();
351                session_metrics::active_sessions(asn).inc();
352
353                self.port_to_client.write().insert(client_addr, port);
354
355                let port = port.into();
356                entry.insert(ClientInfo {
357                    asn_info: asn.cloned(),
358                    created_at: Instant::now(),
359                    port,
360                });
361                Some(port)
362            }
363        }
364    }
365
366    #[inline]
367    fn get_client(&self, port: NetworkU16) -> Option<SocketAddr> {
368        self.port_to_client.read().get(port)
369    }
370}
371
372impl Drop for PortMapper {
373    fn drop(&mut self) {
374        let lock = self.client_to_port.lock();
375
376        let now = Instant::now();
377
378        for client_info in lock.values() {
379            session_metrics::active_sessions(client_info.asn_info.as_ref()).dec();
380            session_metrics::duration_secs()
381                .observe(now.duration_since(client_info.created_at).as_secs_f64());
382        }
383    }
384}

399impl SessionState {
400    /// Attempts to lookup a client endpoint based on the server endpoint that sent
401    /// the packet to the specified port
402    #[inline]
403    fn lookup_client(&self, server_addr: SocketAddr, port: NetworkU16) -> Option<SocketAddr> {
404        self.sessions
405            .get(&server_addr)
406            .and_then(|pm| pm.get_client(port))
407    }
408
409    /// Retrieves the port used to forward packets from the specified client
410    /// endpoint to the specified server endpoint, pairing the port to the client
411    /// for forwarding packets back from the server to the client
412    #[inline]
413    fn get_or_create(
414        &self,
415        client_addr: SocketAddr,
416        server_addr: SocketAddr,
417        asn: Option<&IpNetEntry>,
418    ) -> NetworkU16 {
419        let port = match self.sessions.entry(server_addr) {
420            crate::collections::ttl::Entry::Occupied(entry) => {
421                entry.get().get_or_alloc(client_addr, asn)
422            }
423            crate::collections::ttl::Entry::Vacant(entry) => {
424                let pm = PortMapper::new();
425                let port = pm.get_or_alloc(client_addr, asn);
426                entry.insert(pm);
427                port
428            }
429        };
430
431        if let Some(port) = port {
432            return port;
433        }
434
435        // This means that this server has allocated over 4535 ports, which...could?
436        // happen in some scenarios, but for now we just emit a warning, nuke the current
437        // mapping. This means that if the server is still active and sends packets
438        // in the future, they will either be dropped since we don't know what
439        // the client endpoint is any longer, or, slightly worse, a packet gets
440        // redirected to a different client.
441        self.sessions.remove(server_addr);
442        self.get_or_create(client_addr, server_addr, asn)
443    }
444}

446#[inline]
447pub fn process_packets<const RXN: usize, const TXN: usize>(
448    rx_slab: &mut StackSlab<RXN>,
449    umem: &mut Umem,
450    tx_slab: &mut StackSlab<TXN>,
451    config_state: &mut ConfigState,
452    state: &mut State,
453) {
454    let filters = config_state.filters.load();
455    let cm = config_state.clusters.clone_value();
456
457    let now = UtcTimestamp::now();
458    let jitter = (now - state.last_receive).nanos();
459    state.last_receive = now;
460    let mut had_read = false;
461
462    while let Some(buffer) = rx_slab.pop_back() {
463        let Ok(Some(headers)) = UdpHeaders::parse_packet(&buffer) else {
464            unreachable!(
465                "we somehow got a non-UDP packet, this should be impossible with the eBPF program we use to route packets"
466            );
467        };
468
469        if headers.udp.destination == state.qcmp_port {
470            process_qcmp_packet(buffer, headers, umem, tx_slab);
471            continue;
472        }
473
474        let is_client = headers.udp.destination == state.external_port;
475        let direction = if is_client {
476            had_read = true;
477            metrics::READ
478        } else {
479            metrics::WRITE
480        };
481
482        // This indicates a packet that is split, which we don't handle _at all_ right now
483        if buffer.is_continued() {
484            metrics::packets_dropped_total(direction, "split packet", &metrics::EMPTY).inc();
485            umem.free_packet(buffer);
486            continue;
487        }
488
489        let packet = PacketWrapper { buffer, headers };
490
491        let res = {
492            let _timer = metrics::processing_time(direction).start_timer();
493
494            if is_client {
495                process_client_packet(packet, umem, filters, &cm, state, tx_slab)
496            } else {
497                process_server_packet(packet, umem, filters, state, tx_slab, jitter)
498            }
499        };
500
501        match res {
502            Ok(None) => {}
503            Ok(Some(packet)) => {
504                umem.free_packet(packet);
505            }
506            Err((error, packet)) => {
507                let discriminant = error.discriminant();
508                error.inc_system_errors_total(direction, &metrics::EMPTY);
509                metrics::packets_dropped_total(direction, discriminant, &metrics::EMPTY).inc();
510
511                umem.free_packet(packet);
512            }
513        }
514    }
515
516    if had_read {
517        metrics::packet_jitter(metrics::READ, &metrics::EMPTY).set(jitter);
518    }
519}

521#[inline]
522fn push_packet<const TXN: usize>(
523    direction: metrics::Direction,
524    packet: Packet,
525    asn: AsnInfo<'_>,
526    data_length: usize,
527    res: Result<(), PacketError>,
528    tx_slab: &mut StackSlab<TXN>,
529    umem: &mut Umem,
530) {
531    match res {
532        Ok(()) => {
533            if let Some(packet) = tx_slab.push_front(packet) {
534                metrics::packets_dropped_total(direction, "tx slab full", &metrics::EMPTY).inc();
535                umem.free_packet(packet);
536            } else {
537                metrics::packets_total(direction, &asn).inc();
538                metrics::bytes_total(direction, &asn).inc_by(data_length as u64);
539            }
540        }
541        Err(err) => {
542            let discriminant = err.discriminant();
543            metrics::errors_total(direction, discriminant, &metrics::EMPTY).inc();
544            metrics::packets_dropped_total(direction, discriminant, &metrics::EMPTY).inc();
545            umem.free_packet(packet);
546        }
547    }
548}

550#[inline]
551fn process_client_packet<const TXN: usize>(
552    packet: PacketWrapper,
553    umem: &mut Umem,
554    filters: &filters::FilterChain,
555    cm: &crate::net::ClusterMap,
556    state: &mut State,
557    tx_slab: &mut StackSlab<TXN>,
558) -> Result<Option<Packet>, (PipelineError, Packet)> {
559    let mut source_addr = packet.headers.source_address();
560    source_addr.set_ip(source_addr.ip().to_canonical());
561
562    let mut ctx =
563        filters::ReadContext::new(cm, source_addr.into(), packet, &mut state.destinations);
564
565    let mut packet = match filters.read(&mut ctx) {
566        Ok(()) => ctx.contents,
567        Err(err) => {
568            return Err((PipelineError::Filter(err), ctx.contents.buffer));
569        }
570    };
571
572    let Some(dest_addr) = state.destinations.pop() else {
573        return Ok(Some(packet.buffer));
574    };
575
576    let data = &packet.buffer[packet.headers.data];
577
578    // TODO: We _could_ be more clever with this and do a running checksum calculation
579    // as the packet data is modified by the filters, but for now we just do the
580    // full checksum for the sake of simplicity
581    let data_checksum = csum::DataChecksum::calculate_if_needed(data, &packet.buffer);
582    let data_length = data.len();
583
584    let eth = packet.headers.eth.swapped();
585
586    // If we have more than 1 destination we need to clone the packet data to
587    // a new packet for each destination, only modifying the headers
588    if !state.destinations.is_empty() {
589        while let Some(daddr) = state.destinations.pop() {
590            let Ok(dest_addr) = daddr.to_socket_addr() else {
591                continue;
592            };
593            let (source, asn, ips) = state.session(source_addr, dest_addr);
594
595            let mut headers = UdpHeaders {
596                eth,
597                ip: ips.with_header(&packet.headers.ip),
598                udp: UdpHdr {
599                    source,
600                    destination: dest_addr.port().into(),
601                    check: 0,
602                    length: NetworkU16(0),
603                },
604                data: packet.headers.data,
605            };
606
607            // SAFETY: the umem outlives the frame
608            let mut new_packet = unsafe {
609                let Some(new_packet) = umem.alloc() else {
610                    continue;
611                };
612                new_packet
613            };
614
615            let res = fill_packet(&mut headers, data, data_checksum, &mut new_packet);
616            push_packet(
617                metrics::Direction::Read,
618                new_packet,
619                asn,
620                data_length,
621                res,
622                tx_slab,
623                umem,
624            );
625        }
626    }
627
628    let Ok(dest_addr) = dest_addr.to_socket_addr() else {
629        return Ok(Some(packet.buffer));
630    };
631    let (source, asn, ips) = state.session(source_addr, dest_addr);
632
633    let mut headers = UdpHeaders {
634        eth,
635        ip: ips.with_header(&packet.headers.ip),
636        udp: UdpHdr {
637            source,
638            destination: dest_addr.port().into(),
639            check: 0,
640            length: NetworkU16(0),
641        },
642        data: packet.headers.data,
643    };
644
645    headers.calc_checksum(data_checksum);
646
647    let res = modify_packet_headers(&packet.headers, &mut headers, &mut packet.buffer);
648    push_packet(
649        metrics::Direction::Read,
650        packet.buffer,
651        asn,
652        data_length,
653        res,
654        tx_slab,
655        umem,
656    );
657
658    Ok(None)
659}

661#[inline]
662fn process_server_packet<const TXN: usize>(
663    packet: PacketWrapper,
664    umem: &mut Umem,
665    filters: &crate::filters::FilterChain,
666    state: &mut State,
667    tx_slab: &mut StackSlab<TXN>,
668    jitter: i64,
669) -> Result<Option<Packet>, (PipelineError, Packet)> {
670    let mut server_addr = packet.headers.source_address();
671    server_addr.set_ip(server_addr.ip().to_canonical());
672
673    let Some((client_addr, asn)) = state.lookup_client(server_addr, packet.headers.udp.destination)
674    else {
675        tracing::debug!(address = %server_addr, "received traffic from a server that has no downstream");
676        return Ok(Some(packet.buffer));
677    };
678
679    metrics::packet_jitter(metrics::Direction::Write, &asn).set(jitter);
680
681    let mut ctx = filters::WriteContext::new(server_addr.into(), client_addr.into(), packet);
682
683    let mut packet = match filters.write(&mut ctx) {
684        Ok(()) => ctx.contents,
685        Err(err) => {
686            return Err((PipelineError::Filter(err), ctx.contents.buffer));
687        }
688    };
689
690    let mut headers = UdpHeaders {
691        eth: packet.headers.eth.swapped(),
692        ip: state.ips(client_addr.ip()).with_header(&packet.headers.ip),
693        udp: UdpHdr {
694            source: state.external_port,
695            destination: client_addr.port().into(),
696            length: NetworkU16(0),
697            check: 0,
698        },
699        data: packet.headers.data,
700    };
701
702    let res = modify_packet_headers(&packet.headers, &mut headers, &mut packet.buffer);
703    if res.is_ok() {
704        let _ = packet.buffer.calc_udp_checksum();
705    }
706
707    push_packet(
708        metrics::Direction::Write,
709        packet.buffer,
710        asn,
711        packet.headers.data_length(),
712        res,
713        tx_slab,
714        umem,
715    );
716    Ok(None)
717}

719/// Modifies the headers of an existing well formed packet to a new source and destination,
720/// resizing the header portion as needed if changing between ipv4 and ipv6
721#[inline]
722fn modify_packet_headers(
723    original: &UdpHeaders,
724    new: &mut UdpHeaders,
725    packet: &mut Packet,
726) -> Result<(), PacketError> {
727    match (original.is_ipv4(), new.is_ipv4()) {
728        (true, false) => packet.adjust_head(-20)?,
729        (false, true) => packet.adjust_head(20)?,
730        (_, _) => {}
731    }
732
733    new.set_packet_headers(packet)?;
734    Ok(())
735}
736
737#[inline]
738fn fill_packet(
739    headers: &mut UdpHeaders,
740    data: &[u8],
741    data_checksum: csum::DataChecksum,
742    frame: &mut Packet,
743) -> Result<(), PacketError> {
744    let hdr_len = headers.header_length();
745    frame.adjust_tail(hdr_len as i32)?;
746    headers.calc_checksum(data_checksum);
747    headers.set_packet_headers(frame)?;
748    frame.insert(hdr_len, data)?;
749    Ok(())
750}
751
752fn process_qcmp_packet<const TXN: usize>(
753    mut packet: Packet,
754    headers: UdpHeaders,
755    umem: &mut Umem,
756    tx_slab: &mut StackSlab<TXN>,
757) {
758    use crate::{codec::qcmp, time::UtcTimestamp};
759
760    fn inner(packet: &mut Packet, headers: UdpHeaders) -> bool {
761        let received_at = UtcTimestamp::now();
762        let Some(data) = packet.get(headers.data.start..headers.data.end) else {
763            tracing::debug!("corrupt UDP packet, data payload is out of range");
764            return false;
765        };
766        let command = match qcmp::Protocol::parse(data) {
767            Ok(Some(command)) => command,
768            Ok(None) => {
769                tracing::debug!("rejected non-qcmp packet");
770                return false;
771            }
772            Err(error) => {
773                tracing::debug!(%error, "rejected malformed packet");
774                return false;
775            }
776        };
777
778        let qcmp::Protocol::Ping {
779            client_timestamp,
780            nonce,
781        } = command
782        else {
783            tracing::warn!("rejected unsupported QCMP packet");
784            return false;
785        };
786
787        let mut ob = qcmp::QcmpPacket::default();
788        let buf = qcmp::Protocol::ping_reply(nonce, client_timestamp, received_at).encode(&mut ob);
789
790        if let Err(error) = packet.adjust_tail(-(headers.data_length() as i32)) {
791            tracing::debug!(%error, "unable to trim QCMP ping data");
792            return false;
793        }
794
795        if let Err(error) = packet.insert(headers.data.start, buf) {
796            tracing::debug!(%error, "unable to write QCMP pong data");
797            return false;
798        }
799
800        let mut new = UdpHeaders::new(
801            headers.eth.swapped(),
802            headers.ip.swapped(),
803            headers.udp.swapped(),
804            headers.data.start..headers.data.start + buf.len(),
805        );
806        new.decrement_hop();
807
808        if let Err(error) = modify_packet_headers(&headers, &mut new, packet) {
809            tracing::debug!(%error, "unable to modify QCMP packet headers");
810            return false;
811        }
812
813        if let Err(error) = packet.calc_udp_checksum() {
814            tracing::debug!(%error, "failed to calculate QCMP packet checksum");
815            return false;
816        }
817
818        true
819    }
820
821    let packet = if inner(&mut packet, headers) {
822        tracing::debug!("sending QCMP pong");
823
824        if let Some(packet) = tx_slab.push_front(packet) {
825            tracing::debug!("tx slab full, unable to send QCMP pong");
826            packet
827        } else {
828            return;
829        }
830    } else {
831        packet
832    };
833
834    umem.free_packet(packet);
835}

869        let mut buffer = xdp::Packet::testing_new(&mut data);

871        headers.set_packet_headers(&mut buffer).unwrap();
872        buffer.insert(headers.data.start, &payload).unwrap();
873        buffer.calc_udp_checksum().unwrap();
874
875        let mut wrapper = PacketWrapper { buffer, headers };
876
877        use crate::filters::{Packet, PacketMut};
878
879        assert_eq!(wrapper.as_slice(), payload);
880
881        {
882            const HEAD: &[u8] = &[1; 3];
883            wrapper.extend_head(HEAD);

886            assert_eq!(wrapper.headers.data_length(), payload.len() + HEAD.len());
887            wrapper.remove_head(HEAD.len());
888        }
889
890        assert_eq!(wrapper.as_slice(), payload);
891
892        {
893            const TAIL: &[u8] = &[8; 20];
894            wrapper.extend_tail(TAIL);
895            assert_eq!(wrapper.as_slice()[..payload.len()], payload);
896            assert_eq!(&wrapper.as_slice()[payload.len()..], TAIL);
897            assert_eq!(wrapper.headers.data_length(), payload.len() + TAIL.len());
898            wrapper.remove_tail(TAIL.len());
899        }
