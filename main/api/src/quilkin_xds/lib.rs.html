39/// Checks if the status is a "broken pipe" eg. the other end of the connection
40/// was terminated.. THIS IS NOT AN ERROR
41fn is_broken_pipe(err_status: &tonic::Status) -> bool {
42    use std::error::Error as _;
43    let mut error = err_status.source();
44
45    'lup: while let Some(err) = error {
46        let io = 'block: {
47            if let Some(io_err) = err.downcast_ref::<std::io::Error>() {
48                break 'block io_err;
49            }
50
51            // h2::Error do not expose std::io::Error with `source()`
52            // https://github.com/hyperium/h2/pull/462
53            if let Some(h2_err) = err.downcast_ref::<h2::Error>()
54                && let Some(io_err) = h2_err.get_io()
55            {
56                break 'block io_err;
57            }
58
59            error = err.source();
60            continue 'lup;
61        };
62
63        return io.kind() == std::io::ErrorKind::BrokenPipe;
64    }
65
66    false
67}
